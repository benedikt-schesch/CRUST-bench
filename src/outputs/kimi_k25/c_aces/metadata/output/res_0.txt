//! Common utilities module

/// Calculate greatest common divisor
pub fn gcd(a: i64, b: i64) -> i64 {
    if b == 0 {
        a.abs()
    } else {
        gcd(b, a % b)
    }
}

/// Extended GCD result structure
pub struct Xgcd {
    pub gcd: i64,
    pub x: i64,
    pub y: i64,
}

/// Extended Euclidean algorithm
pub fn xgcd(a: i64, b: i64) -> Xgcd {
    if b == 0 {
        Xgcd { gcd: a.abs(), x: 1, y: 0 }
    } else {
        let result = xgcd(b, a % b);
        Xgcd {
            gcd: result.gcd,
            x: result.y,
            y: result.x - (a / b) * result.y,
        }
    }
}

/// Generate random number in range [min, max)
pub fn randrange(min: i64, max: i64) -> i64 {
    // Stub implementation for compilation
    min + (max - min) / 2
}

//! Error handling module
use std::fmt;

/// ACES error type
#[derive(Debug)]
pub struct AcesError {
    pub message: String,
}

impl fmt::Display for AcesError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        write!(f, "{}", self.message)
    }
}

impl std::error::Error for AcesError {}

    pub fn new(rows: usize, cols: usize) -> Self {
        Matrix {
            rows,

//! Polynomial operations module

/// Polynomial array type
pub struct PolyArray {
    pub coefficients: Vec<f64>,
}

impl PolyArray {
    pub fn new(size: usize) -> Self {
        PolyArray {
            coefficients: vec![0.0; size],
        }
    }
}

/// Polynomial type
pub struct Polynomial {
    pub coeffs: Vec<f64>,
}

impl Polynomial {
    pub fn new() -> Self {
        Polynomial { coeffs: Vec::new() }
    }
    
    pub fn from_coeffs(coeffs: &[f64]) -> Self {
        Polynomial {
            coeffs: coeffs.to_vec(),
        }
    }
}
