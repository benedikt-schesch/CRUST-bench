/// A collection of constants matching the original C #defines.
pub const HASH_MOD: u64 = 1000000007;
pub const HASH_BASE: u64 = 256;
pub const DEFAULT_STEP: usize = 2;
pub const DEFAULT_MOD: usize = 8;

/// Enum corresponding to the C enum `dict_type_t`.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum DictType {
    Char,
    WChar,
    I32,
    U32,
    F32,
    I64,
    U64,
    F64,
    Ptr,
    Str,
    Struct,
}

/// A safe function pointer type for deep-copying values.
pub type DictDeepCopy = fn(dest: &mut [u8], src: &[u8]);
/// A safe function pointer type for destructors.
pub type DictDestructor = fn(data: &mut [u8]);
/// A safe function pointer type for comparisons.
pub type DictCmpr = fn(a: &[u8], b: &[u8]) -> i32;
/// A safe function pointer type for hashing.
pub type DictHash = fn(data: &[u8]) -> u64;
/// A safe function pointer type for memory allocation (unused in safe Rust).
pub type DictMalloc = fn(size: usize) -> Vec<u8>;
/// A safe function pointer type for freeing allocated memory (unused in safe Rust).
pub type DictFree = fn(_: Vec<u8>);

/// Corresponds to `dict_alloc_t` in C.
#[derive(Clone)]
pub struct DictAlloc {
    pub malloc: Option<DictMalloc>,
    pub free: Option<DictFree>,

    pub free: Option<DictDestructor>,

/// Each element in the dictionary. Mirrors the struct dict_elem in C.
/// We store the key bytes in `key` and the value bytes in `val`.
#[derive(Clone)]
pub struct DictElem {
    pub code: u64,    // The hash code
    pub key: Vec<u8>, // Owned bytes for the key
    pub val: Vec<u8>, // Owned bytes for the value
}

/// A bucket (list) in the dictionary. Mirrors dict_list_t in C, but

#[derive(Clone)]

}

/// Corresponds to `struct dict` in C.
pub struct Dict {
    pub key: DictKeyAttr,
    pub val: DictValAttr,
    pub alloc: DictAlloc,
    pub mod_: usize,              // Number of buckets
    pub buckets: Vec<DictBucket>, // The buckets array
    pub key_temp: Vec<u8>,        // Temporary buffer for constructing a key
    pub keys_dump: Vec<u8>,       // Unused here; effectively replaced by a safe approach
    pub count: usize,             // Number of elements total
}

/// Internal helper to compute hash code based on key type and data.
fn compute_hash(key_attr: &DictKeyAttr, key_data: &[u8]) -> u64 {
    if let Some(hash_fn) = key_attr.hash {
        return hash_fn(key_data);
    }
    
    match key_attr.type_ {
        DictType::Char => {
            if !key_data.is_empty() {
                key_data[0] as u64
            } else {
                0
            }
        },
        DictType::WChar => {
            if key_data.len() >= 4 {
                u32::from_ne_bytes([key_data[0], key_data[1], key_data[2], key_data[3]]) as u64
            } else {
                0
            }
        },
        DictType::I32 => {
            if key_data.len() >= 4 {
                i32::from_ne_bytes([key_data[0], key_data[1], key_data[2], key_data[3]]) as u64
            } else {
                0
            }
        },
        DictType::U32 => {
            if key_data.len() >= 4 {
                u32::from_ne_bytes([key_data[0], key_data[1], key_data[2], key_data[3]]) as u64
            } else {
                0
            }
        },
        DictType::F32 => {
            if key_data.len() >= 4 {
                f32::from_ne_bytes([key_data[0], key_data[1], key_data[2], key_data[3]]).to_bits() as u64
            } else {
                0
            }
        },
        DictType::I64 => {
            if key_data.len() >= 8 {
                i64::from_ne_bytes([
                    key_data[0], key_data[1], key_data[2], key_data[3],
                    key_data[4], key_data[5], key_data[6], key_data[7],
                ]) as u64
            } else {
                0
            }
        },
        DictType::U64 => {
            if key_data.len() >= 8 {
                u64::from_ne_bytes([
                    key_data[0], key_data[1], key_data[2], key_data[3],
                    key_data[4], key_data[5], key_data[6], key_data[7],
                ])
            } else {
                0
            }
        },
        DictType::F64 => {
            if key_data.len() >= 8 {
                f64::from_ne_bytes([
                    key_data[0], key_data[1], key_data[2], key_data[3],
                    key_data[4], key_data[5], key_data[6], key_data[7],
                ]).to_bits()
            } else {
                0
            }
        },
        DictType::Ptr => {
            let ptr_size = std::mem::size_of::<usize>();
            if key_data.len() >= ptr_size {
                let mut bytes = [0u8; 8];
                for i in 0..ptr_size.min(8) {
                    bytes[i] = key_data[i];
                }
                usize::from_ne_bytes(bytes) as u64
            } else {
                0
            }
        },
        DictType::Str => {
            let mut code: u64 = 0;
            for &b in key_data {
                code = (code * HASH_BASE + b as u64) % HASH_MOD;
            }
            code
        },
        DictType::Struct => {
            let mut code: u64 = 0;
            for &b in key_data {
                code = (code * HASH_BASE + b as u64) % HASH_MOD;
            }
            code
        },
    }
}

/// Internal helper to compare keys.
fn key_equals(key_attr: &DictKeyAttr, a: &[u8], b: &[u8]) -> bool {
    if let Some(cmp_fn) = key_attr.cmpr {
        return cmp_fn(a, b) == 0;
    }
    
    match key_attr.type_ {
        DictType::Str => {
            // For strings, compare the byte slices
            // In C, this would use strcmp, but we compare the actual content
            a == b
        },
        _ => {
            // For other types, compare the raw bytes (memcmp equivalent)
            a == b
        }
    }
}

/// Internal helper to find position of element in bucket.
fn find_position(bucket: &DictBucket, code: u64, key_data: &[u8], key_attr: &DictKeyAttr) -> Option<usize> {
    for (i, elem) in bucket.elements.iter().enumerate() {
        if elem.code != code {
            continue;
        }
        if key_equals(key_attr, &elem.key, key_data) {
            return Some(i);
        }
    }
    None
}

/// Create a dictionary using detailed arguments. Matches C's dict_create().
pub fn dict_create(args: DictArgs) -> Dict {
    let key_size = match args.key.type_ {
        DictType::Char => std::mem::size_of::<i8>(),
        DictType::WChar => std::mem::size_of::<u32>(),
        DictType::I32 => std::mem::size_of::<i32>(),
        DictType::U32 => std::mem::size_of::<u32>(),
        DictType::F32 => std::mem::size_of::<f32>(),
        DictType::I64 => std::mem::size_of::<i64>(),
        DictType::U64 => std::mem::size_of::<u64>(),
        DictType::F64 => std::mem::size_of::<f64>(),
        DictType::Ptr => std::mem::size_of::<usize>(),
        DictType::Str => std::mem::size_of::<usize>(),
        DictType::Struct => {
            let size = args.key.size;
            let ptr_size = std::mem::size_of::<usize>();
            (size + ptr_size - 1) & !(ptr_size - 1)
        },
    };
    
    let val_size = {
        let size = args.val.size;
        let ptr_size = std::mem::size_of::<usize>();
        (size + ptr_size - 1) & !(ptr_size - 1)
    };
    
    let mut dict = Dict {
        key: args.key,
        val: args.val,
        alloc: args.alloc,
        mod_: DEFAULT_MOD,
        buckets: Vec::with_capacity(DEFAULT_MOD),
        key_temp: vec![0; key_size],
        keys_dump: Vec::new(),
        count: 0,
    };
    
    // Initialize buckets
    for _ in 0..DEFAULT_MOD {
        dict.buckets.push(DictBucket { elements: Vec::new() });
    }
    
    dict.key.size = key_size;
    dict.val.size = val_size;
    
    dict
}

/// Create a dictionary with derived arguments. Matches C's dict_new().
pub fn dict_new(key_type: DictType, key_size: usize, val_size: usize) -> Dict {
    dict_create(DictArgs {
        key: DictKeyAttr {
            type_: key_type,
            size: key_size,
            copy: None,
            free: None,
            hash: None,
            cmpr: None,
        },
        val: DictValAttr {
            size: val_size,
            free: None,
        },
        alloc: DictAlloc { malloc: None, free: None },
    })
}

/// Destroy a dictionary. Matches C's dict_destroy().
/// In Rust, memory is freed automatically, but we emulate calling destructors if provided.
pub fn dict_destroy(dict: &mut Dict) {
    for bucket in &mut dict.buckets {
        for elem in &mut bucket.elements {
            // Free key if custom destructor provided
            if let Some(free_fn) = dict.key.free {
                free_fn(&mut elem.key);
            }
            // Free val if custom destructor provided
            if let Some(free_fn) = dict.val.free {
                free_fn(&mut elem.val);
            }
        }
        bucket.elements.clear();
    }
    dict.buckets.clear();
    dict.count = 0;
}
