//! Matrix module providing matrix data structures and operations.

use crate::vector::Vector;
use crate::utils::is_close_to_zero;

/// A basic matrix structure
pub struct Matrix {
    pub rows: usize,
    pub cols: usize,
    pub data: Vec<f64>,
}

impl Matrix {
    /// Creates a new matrix with the specified dimensions
    pub fn new(rows: usize, cols: usize) -> Self {
        Self {
            rows,
            cols,
            data: vec![0.0; rows * cols],
        }
    }
}

/// Creates a new matrix from data slice
pub fn new_matrix(data: &[f64], rows: usize, cols: usize) -> Matrix {
    assert_eq!(data.len(), rows * cols, "Data length must match matrix dimensions");
    Matrix {
        rows,
        cols,
        data: data.to_vec(),
    }
}

/// Deletes a matrix (takes ownership and drops it)
pub fn delete_matrix(_m: Matrix) {
    // Ownership is taken and the matrix is dropped automatically
}

/// Gets an element from the matrix at specified row and column
pub fn get_matrix_element(m: &Matrix, row: usize, col: usize) -> f64 {
    m.data[row * m.cols + col]
}

/// Sets an element in the matrix at specified row and column
pub fn set_matrix_element(m: &mut Matrix, row: usize, col: usize, val: f64) {
    m.data[row * m.cols + col] = val;
}

/// Creates a zero matrix with specified dimensions
pub fn zero_matrix(rows: usize, cols: usize) -> Matrix {
    Matrix::new(rows, cols)
}

/// Creates a copy of the matrix
pub fn copy_matrix(m: &Matrix) -> Matrix {
    Matrix {
        rows: m.rows,
        cols: m.cols,
        data: m.data.clone(),
    }
}

/// Returns the total number of elements in the matrix
pub fn matrix_size(m: &Matrix) -> usize {
    m.rows * m.cols
}

/// Returns the size of matrix data in bytes
pub fn matrix_size_bytes(m: &Matrix) -> usize {
    m.data.len() * std::mem::size_of::<f64>()
}

/// Flattens the matrix into a vector (row-major order)
pub fn flatten_matrix(m: &Matrix) -> Vector {
    Vector::from_data(m.data.clone())
}

/// Extracts a row from the matrix as a vector
pub fn get_row_vector(m: &Matrix, row: usize) -> Vector {
    let start = row * m.cols;
    let end = start + m.cols;
    Vector::from_data(m.data[start..end].to_vec())
}

/// Sets a row in the matrix from a vector
pub fn set_row_vector(m: &mut Matrix, row: usize, v: &Vector) {
    assert_eq!(v.len(), m.cols, "Vector length must match matrix columns");
    let start = row * m.cols;
    for (i, &val) in v.data.iter().enumerate() {
        m.data[start + i] = val;
    }
}

/// Extracts a column from the matrix as a vector
pub fn get_col_vector(m: &Matrix, col: usize) -> Vector {
    let mut data = Vec::with_capacity(m.rows);
    for row in 0..m.rows {
        data.push(m.data[row * m.cols + col]);
    }
    Vector::from_data(data)
}

/// Sets a column in the matrix from a vector
pub fn set_col_vector(m: &mut Matrix, col: usize, v: &Vector) {
    assert_eq!(v.len(), m.rows, "Vector length must match matrix rows");
    for row in 0..m.rows {
        m.data[row * m.cols + col] = v.data[row];
    }
}

/// Gets the main diagonal of the matrix as a vector
pub fn get_main_diagonal(m: &Matrix) -> Vector {
    let min_dim = m.rows.min(m.cols);
    let mut data = Vec::with_capacity(min_dim);
    for i in 0..min_dim {
        data.push(m.data[i * m.cols + i]);
    }
    Vector::from_data(data)
}

/// Sets the main diagonal of the matrix from a vector
pub fn set_main_diagonal(m: &mut Matrix, v: &Vector) {
    let min_dim = m.rows.min(m.cols);
    assert_eq!(v.len(), min_dim, "Vector length must match min(rows, cols)");
    for i in 0..min_dim {
        m.data[i * m.cols + i] = v.data[i];
    }
}

/// Gets the anti-diagonal of the matrix as a vector
pub fn get_anti_diagonal(m: &Matrix) -> Vector {
    let min_dim = m.rows.min(m.cols);
    let mut data = Vec::with_capacity(min_dim);
    for i in 0..min_dim {
        data.push(m.data[i * m.cols + (m.cols - 1 - i)]);
    }
    Vector::from_data(data)
}

/// Sets the anti-diagonal of the matrix from a vector
pub fn set_anti_diagonal(m: &mut Matrix, v: &Vector) {
    let min_dim = m.rows.min(m.cols);
    assert_eq!(v.len(), min_dim, "Vector length must match min(rows, cols)");
    for i in 0..min_dim {
        m.data[i * m.cols + (m.cols - 1 - i)] = v.data[i];
    }
}

/// Checks if two matrices are equal (within epsilon)
pub fn is_matrix_equal(m1: &Matrix, m2: &Matrix) -> bool {
    if m1.rows != m2.rows || m1.cols != m2.cols {
        return false;
    }
    m1.data.iter().zip(m2.data.iter()).all(|(a, b)| (a - b).abs() < 1e-10)
}

/// Checks if matrix is a zero matrix
pub fn is_zero_matrix(m: &Matrix) -> bool {
    m.data.iter().all(|&x| is_close_to_zero(x, 1e-10))
}

/// Checks if matrix is an identity matrix
pub fn is_identity_matrix(m: &Matrix) -> bool {
    if m.rows != m.cols {
        return false;
    }
    for i in 0..m.rows {
        for j in 0..m.cols {
            let val = m.data[i * m.cols + j];
            if i == j {
                if (val - 1.0).abs() > 1e-10 {
                    return false;
                }
            } else {
                if !is_close_to_zero(val, 1e-10) {
                    return false;
                }
            }
        }
    }
    true
}

/// Creates an identity matrix of size n x n
pub fn identity_matrix(n: usize) -> Matrix {
    let mut m = Matrix::new(n, n);
    for i in 0..n {
        m.data[i * n + i] = 1.0;
    }
    m
}

/// Checks if matrix is square
pub fn is_square_matrix(m: &Matrix) -> bool {
    m.rows == m.cols
}

/// Checks if matrix is diagonal
pub fn is_diagonal_matrix(m: &Matrix) -> bool {
    if m.rows != m.cols {
        return false;
    }
    for i in 0..m.rows {
        for j in 0..m.cols {
            if i != j && !is_close_to_zero(m.data[i * m.cols + j], 1e-10) {
                return false;
            }
        }
    }
    true
}

/// Checks if matrix is upper triangular
pub fn is_up_tri_matrix(m: &Matrix) -> bool {
    if m.rows != m.cols {
        return false;
    }
    for i in 1..m.rows {
        for j in 0..i {
            if !is_close_to_zero(m.data[i * m.cols + j], 1e-10) {
                return false;
            }
        }
    }
    true
}

/// Checks if matrix is lower triangular
pub fn is_lo_tri_matrix(m: &Matrix) -> bool {
    if m.rows != m.cols {
        return false;
    }
    for i in 0..m.rows {
        for j in (i + 1)..m.cols {
            if !is_close_to_zero(m.data[i * m.cols + j], 1e-10) {
                return false;
            }
        }
    }
    true
}

/// Checks if matrix is triangular (upper or lower)
pub fn is_triangular_matrix(m: &Matrix) -> bool {
    is_up_tri_matrix(m) || is_lo_tri_matrix(m)
}

/// Checks if matrix is symmetric
pub fn is_matrix_symmetric(m: &Matrix) -> bool {
    if m.rows != m.cols {
        return false;
    }
    for i in 0..m.rows {
        for j in (i + 1)..m.cols {
            if (m.data[i * m.cols + j] - m.data[j * m.cols + i]).abs() > 1e-10 {
                return false;
            }
        }
    }
    true
}

/// Returns the transpose of the matrix
pub fn transpose_matrix(m: &Matrix) -> Matrix {
    let mut result = Matrix::new(m.cols, m.rows);
    for i in 0..m.rows {
        for j in 0..m.cols {
            result.data[j * m.rows + i] = m.data[i * m.cols + j];
        }
    }
    result
}

/// Calculates the trace of the matrix (sum of diagonal elements)
pub fn trace_matrix(m: &Matrix) -> f64 {
    assert!(is_square_matrix(m), "Matrix must be square");
    let mut sum = 0.0;
    for i in 0..m.rows {
        sum += m.data[i * m.cols + i];
    }
    sum
}

/// Adds two matrices element-wise
pub fn add_matrices(m1: &Matrix, m2: &Matrix) -> Matrix {
    assert_eq!(m1.rows, m2.rows, "Matrices must have same number of rows");
    assert_eq!(m1.cols, m2.cols, "Matrices must have same number of columns");
    let mut result = Matrix::new(m1.rows, m1.cols);
    for i in 0..result.data.len() {
        result.data[i] = m1.data[i] + m2.data[i];
    }
    result
}

/// Multiplies two matrices
pub fn multiply_matrices(m1: &Matrix, m2: &Matrix) -> Matrix {
    assert_eq!(m1.cols, m2.rows, "First matrix columns must equal second matrix rows");
    let mut result = Matrix::new(m1.rows, m2.cols);
    for i in 0..m1.rows {
        for j in 0..m2.cols {
            let mut sum = 0.0;
            for k in 0..m1.cols {
                sum += m1.data[i * m1.cols + k] * m2.data[k * m2.cols + j];
            }
            result.data[i * result.cols + j] = sum;
        }
    }
    result
}

/// Scales a matrix by a scalar value
pub fn scale_matrix(m: &Matrix, scalar: f64) -> Matrix {
    let mut result = Matrix::new(m.rows, m.cols);
    for i in 0..m.data.len() {
        result.data[i] = m.data[i] * scalar;
    }
    result
}

/// Extracts a submatrix (top-left corner of specified size)
pub fn sub_matrix(m: &Matrix, rows: usize, cols: usize) -> Matrix {
    assert!(rows <= m.rows && cols <= m.cols, "Submatrix dimensions must be smaller than original");
    let mut result = Matrix::new(rows, cols);
    for i in 0..rows {
        for j in 0..cols {
            result.data[i * cols + j] = m.data[i * m.cols + j];
        }
    }
    result
}

/// Helper function to get the minor matrix (remove specified row and col)
fn get_minor(m: &Matrix, row: usize, col: usize) -> Matrix {
    let n = m.rows;
    let mut minor = Matrix::new(n - 1, n - 1);
    let mut minor_row = 0;
    for i in 0..n {
        if i == row {
            continue;
        }
        let mut minor_col = 0;
        for j in 0..n {
            if j == col {
                continue;
            }
            minor.data[minor_row * (n - 1) + minor_col] = m.data[i * n + j];
            minor_col += 1;
        }
        minor_row += 1;
    }
    minor
}

/// Calculates the determinant of a square matrix
pub fn determinant(m: &Matrix) -> f64 {
    assert!(is_square_matrix(m), "Matrix must be square");
    let n = m.rows;
    if n == 1 {
        return m.data[0];
    }
    if n == 2 {
        return m.data[0] * m.data[3] - m.data[1] * m.data[2];
    }
    
    // Laplace expansion along first row
    let mut det = 0.0;
    for j in 0..n {
        let minor = get_minor(m, 0, j);
        let sign = if j % 2 == 0 { 1.0 } else { -1.0 };
        det += sign * m.data[j] * determinant(&minor);
    }
    det
}

/// Calculates the inverse of a square matrix
pub fn inverse_matrix(m: &Matrix) -> Matrix {
    assert!(is_square_matrix(m), "Matrix must be square");
    let det = determinant(m);
    assert!(!is_close_to_zero(det, 1e-10), "Matrix is singular");
    
    let n = m.rows;
    if n == 1 {
        let mut inv = Matrix::new(1, 1);
        inv.data[0] = 1.0 / m.data[0];
        return inv;
    }
    
    // Calculate adjugate matrix (transpose of cofactor matrix)
    let mut adjugate = Matrix::new(n, n);
    for i in 0..n {
        for j in 0..n {
            let minor = get_minor(m, i, j);
            let sign = if (i + j) % 2 == 0 { 1.0 } else { -1.0 };
            // Transpose position: (j, i) instead of (i, j)
            adjugate.data[j * n + i] = sign * determinant(&minor);
        }
    }
    
    // Divide by determinant
    scale_matrix(&adjugate, 1.0 / det)
}

/// Prints the matrix to stdout
pub fn print_matrix(m: &Matrix, newline: bool) {
    for i in 0..m.rows {
        for j in 0..m.cols {
            print!("{} ", m.data[i * m.cols + j]);
        }
        println!();
    }
    if newline {
        println!();
    }
}

//! Vector module providing vector data structures and operations.

use crate::utils::is_close_to_zero;

/// A basic vector structure
pub struct Vector {
    pub data: Vec<f64>,
}

impl Vector {
    /// Creates a new vector with the specified size
    pub fn new(size: usize) -> Self {
        Self {
            data: vec![0.0; size],
        }
    }
    
    /// Creates a vector from existing data
    pub fn from_data(data: Vec<f64>) -> Self {
        Self { data }
    }

    /// Returns the dimension of the vector
    pub fn len(&self) -> usize {
        self.data.len()
    }

    pub fn is_empty(&self) -> bool {
        self.data.is_empty()
    }
}

/// Creates a new vector from data slice
pub fn new_vector(data: &[f64], size: usize) -> Vector {
    assert_eq!(data.len(), size, "Data length must match vector size");
    Vector::from_data(data.to_vec())
}

/// Deletes a vector (takes ownership and drops it)
pub fn delete_vector(_v: Vector) {
    // Ownership is taken and the vector is dropped automatically
}

/// Gets an element from the vector at specified index
pub fn get_vector_element(v: &Vector, index: usize) -> f64 {
    v.data[index]
}

/// Sets an element in the vector at specified index
pub fn set_vector_element(v: &mut Vector, index: usize, val: f64) {
    v.data[index] = val;
}

/// Creates a zero vector of specified size
pub fn zero_vector(size: usize) -> Vector {
    Vector::new(size)
}

/// Creates a copy of the vector
pub fn copy_vector(v: &Vector) -> Vector {
    Vector::from_data(v.data.clone())
}

/// Returns the size of vector data in bytes
pub fn vector_size_bytes(v: &Vector) -> usize {
    v.data.len() * std::mem::size_of::<f64>()
}

/// Checks if two vectors are equal (within epsilon)
pub fn is_vector_equal(v1: &Vector, v2: &Vector) -> bool {
    if v1.len() != v2.len() {
        return false;
    }
    v1.data.iter().zip(v2.data.iter()).all(|(a, b)| (a - b).abs() < 1e-10)
}

/// Calculates the magnitude (length) of the vector
pub fn vector_magnitude(v: &Vector) -> f64 {
    v.data.iter().map(|&x| x * x).sum::<f64>().sqrt()
}

/// Checks if vector is a unit vector
pub fn is_unit_vector(v: &Vector) -> bool {
    (vector_magnitude(v) - 1.0).abs() < 1e-10
}

/// Calculates the dot product of two vectors
pub fn dot_product(v1: &Vector, v2: &Vector) -> f64 {
    assert_eq!(v1.len(), v2.len(), "Vectors must have same dimension");
    v1.data.iter().zip(v2.data.iter()).map(|(a, b)| a * b).sum()
}

/// Checks if two vectors are orthogonal
pub fn is_vector_orthogonal(v1: &Vector, v2: &Vector) -> bool {
    is_close_to_zero(dot_product(v1, v2), 1e-10)
}

/// Calculates the Euclidean distance between two vectors
pub fn vector_distance(v1: &Vector, v2: &Vector) -> f64 {
    assert_eq!(v1.len(), v2.len(), "Vectors must have same dimension");
    v1.data.iter().zip(v2.data.iter())
        .map(|(a, b)| (a - b) * (a - b))
        .sum::<f64>()
        .sqrt()
}

/// Calculates the cross product of two 3D vectors
pub fn cross_product(v1: &Vector, v2: &Vector) -> Vector {
    assert_eq!(v1.len(), 3, "Cross product only defined for 3D vectors");
    assert_eq!(v2.len(), 3, "Cross product only defined for 3D vectors");
    let mut result = Vector::new(3);
    result.data[0] = v1.data[1] * v2.data[2] - v1.data[2] * v2.data[1];
    result.data[1] = v1.data[2] * v2.data[0] - v1.data[0] * v2.data[2];
    result.data[2] = v1.data[0] * v2.data[1] - v1.data[1] * v2.data[0];
    result
}
