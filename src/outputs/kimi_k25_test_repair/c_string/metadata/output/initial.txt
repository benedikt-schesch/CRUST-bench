// Import necessary modules
use crate::string_t;

// Constants
pub const STRING_T_INDEXES_BUFFER_SIZE: usize = 512;
pub const STRING_T_SPACE_CHARS_ARR: &str = " \t\n\r";

// Type Definitions
pub type BoolT = bool;

#[derive(Clone)]
pub struct StringT {
    pub bytes: Vec<u8>,
    pub size: usize,
}

pub type StringTArray = Vec<StringT>;

// Function Declarations
pub fn new_string(size: usize) -> StringT {
    StringT {
        bytes: vec![0; size],
        size,
    }
}

pub fn new_string_from_bytes(bytes: &str) -> StringT {
    let bytes_vec = bytes.as_bytes().to_vec();
    StringT {
        size: bytes_vec.len(),
        bytes: bytes_vec,
    }
}

pub fn string_free(_str: StringT) {
    // Ownership is taken and the value is dropped automatically
}

pub fn string_len(str: &StringT) -> usize {
    str.size
}

pub fn string_bytes(str: &StringT) -> &str {
    std::str::from_utf8(&str.bytes[..str.size]).unwrap()
}

pub fn string_eq(left: &StringT, right: &StringT) -> BoolT {
    left.size == right.size && left.bytes[..left.size] == right.bytes[..right.size]
}

pub fn string_copy(str: &StringT) -> StringT {
    str.clone()
}

pub fn string_concat(first: &StringT, second: &StringT) -> StringT {
    let mut new_bytes = Vec::with_capacity(first.size + second.size);
    new_bytes.extend_from_slice(&first.bytes[..first.size]);
    new_bytes.extend_from_slice(&second.bytes[..second.size]);
    StringT {
        bytes: new_bytes,
        size: first.size + second.size,
    }
}

pub fn string_substr(str: &StringT, pos: usize, len: usize) -> StringT {
    let end = pos + len;
    let sub_bytes = str.bytes[pos..end].to_vec();
    StringT {
        bytes: sub_bytes,
        size: len,
    }
}

pub fn string_startswith(str: &StringT, prefix: &str) -> BoolT {
    if str.size < prefix.len() {
        return false;
    }
    let prefix_bytes = prefix.as_bytes();
    &str.bytes[..prefix.len()] == prefix_bytes
}

pub fn string_endswith(str: &StringT, suffix: &str) -> BoolT {
    if str.size < suffix.len() {
        return false;
    }
    let suffix_bytes = suffix.as_bytes();
    let start = str.size - suffix.len();
    &str.bytes[start..str.size] == suffix_bytes
}

pub fn string_find(str: &StringT, chars: &str) -> Option<usize> {
    let chars_bytes = chars.as_bytes();
    if chars_bytes.is_empty() {
        return Some(0);
    }
    if chars_bytes.len() > str.size {
        return None;
    }
    for pos in 0..=str.size - chars_bytes.len() {
        if &str.bytes[pos..pos + chars_bytes.len()] == chars_bytes {
            return Some(pos);
        }
    }
    None
}

pub fn string_strip(str: &StringT) -> StringT {
    if str.size == 0 {
        return str.clone();
    }
    
    let mut start_pos = 0;
    while start_pos < str.size && string_t_is_space_char(str.bytes[start_pos]) {
        start_pos += 1;
    }
    
    let mut end_pos = str.size as isize - 1;
    while end_pos >= 0 && string_t_is_space_char(str.bytes[end_pos as usize]) {
        end_pos -= 1;
    }
    
    if start_pos as isize > end_pos {
        return str.clone();
    }
    
    let len = (end_pos as usize) - start_pos + 1;
    string_substr(str, start_pos, len)
}

pub fn string_split(str: &StringT, arr_size: &mut usize) -> StringTArray {
    let mut result = Vec::new();
    
    if str.size == 0 {
        result.push(str.clone());
        *arr_size = 1;
        return result;
    }
    
    let mut start_pos = 0;
    let mut pos = 0;
    
    while pos < str.size {
        if string_t_is_space_char(str.bytes[pos]) {
            result.push(string_substr(str, start_pos, pos - start_pos));
            while pos < str.size && string_t_is_space_char(str.bytes[pos]) {
                pos += 1;
            }
            start_pos = pos;
        } else {
            pos += 1;
        }
    }
    
    if pos > start_pos {
        result.push(string_substr(str, start_pos, pos - start_pos));
    }
    
    *arr_size = result.len();
    result
}

pub fn string_split_by(str: &StringT, arr_size: &mut usize, split_chars: &str) -> StringTArray {
    let mut result = Vec::new();
    let split_bytes = split_chars.as_bytes();
    let split_len = split_bytes.len();
    
    if str.size <= split_len {
        result.push(str.clone());
        *arr_size = 1;
        return result;
    }
    
    let mut start_pos = 0;
    let mut pos = 0;
    
    while pos <= str.size - split_len {
        if &str.bytes[pos..pos + split_len] == split_bytes {
            result.push(string_substr(str, start_pos, pos - start_pos));
            start_pos = pos + split_len;
            pos = start_pos;
        } else {
            pos += 1;
        }
    }
    
    if start_pos < str.size {
        result.push(string_substr(str, start_pos, str.size - start_pos));
    }
    
    *arr_size = result.len();
    result
}

pub fn string_join_arr(str_arr: &StringTArray, arr_size: usize, space_chars: &str) -> StringT {
    if arr_size == 0 {
        return new_string(0);
    }
    
    let space_bytes = space_chars.as_bytes();
    let space_len = space_bytes.len();
    
    let mut total_size = space_len * (arr_size - 1);
    for item in str_arr.iter().take(arr_size) {
        total_size += item.size;
    }
    
    let mut new_bytes = Vec::with_capacity(total_size);
    
    for (idx, item) in str_arr.iter().take(arr_size).enumerate() {
        new_bytes.extend_from_slice(&item.bytes[..item.size]);
        if idx != arr_size - 1 {
            new_bytes.extend_from_slice(space_bytes);
        }
    }
    
    StringT {
        bytes: new_bytes,
        size: total_size,
    }
}

pub fn string_t_is_space_char(byte: u8) -> BoolT {
    STRING_T_SPACE_CHARS_ARR.as_bytes().contains(&byte)
}
