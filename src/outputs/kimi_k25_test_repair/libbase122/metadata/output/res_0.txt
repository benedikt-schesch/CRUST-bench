pub struct BitReader<'a> {
    pub input: &'a [u8],
    pub byte_pos: usize,
    pub bit_pos: usize,
}
impl<'a> BitReader<'a> {
    pub fn new(input: &'a [u8]) -> Self {
        BitReader {
            input,
            byte_pos: 0,
            bit_pos: 0,
        }
    }
    pub fn read(&mut self, nbits: u8) -> (u8, u8) {
        let total_bits = self.input.len() * 8;
        let current_bit = self.byte_pos * 8 + self.bit_pos;
        let max_nbits = total_bits.saturating_sub(current_bit);
        let nbits = if (nbits as usize) > max_nbits {
            max_nbits as u8
        } else {
            nbits
        };
        if nbits == 0 {
            return (0, 0);
        }
        let first_byte_index = self.byte_pos;
        let first_byte_cur_bit = self.bit_pos;
        let first_byte = self.input[first_byte_index] as u16;
        let second_byte = if first_byte_index + 1 < self.input.len() {
            self.input[first_byte_index + 1] as u16
        } else {
            0
        };
        let mut two_bytes = (first_byte << 8) | second_byte;
        let shift = (8 - first_byte_cur_bit) + (8 - nbits as usize);
        two_bytes = two_bytes >> shift;
        let mask = !(0xFFu16 << nbits);
        let result = (two_bytes & mask) as u8;
        let bits_read = nbits as usize;
        self.bit_pos += bits_read;
        self.byte_pos += self.bit_pos / 8;
        self.bit_pos %= 8;
        (nbits, result)
    }
}
pub struct BitWriter<'a> {
    pub output: Option<&'a mut [u8]>,
    pub len: usize,
    pub cur_bit: usize,
    pub count_only: bool,
}
impl<'a> BitWriter<'a> {
    pub fn new(output: Option<&'a mut [u8]>, len: usize) -> Self {
        let count_only = output.is_none();
        BitWriter {
            output,
            len,
            cur_bit: 0,
            count_only,
        }
    }
    pub fn write(&mut self, nbits: u8, value: u8) -> Result<usize, Base122Error> {
        if self.count_only {
            self.cur_bit += nbits as usize;
            return Ok(self.cur_bit / 8);
        }
        if self.cur_bit + nbits as usize > self.len * 8 {
            return Err(Base122Error::new("Output does not have sufficient size"));
        }
        let first_byte_index = self.cur_bit / 8;
        let first_byte_cur_bit = self.cur_bit % 8;
        let mask1 = !(0xFFu8 >> first_byte_cur_bit);
        let existing = self.output.as_mut().unwrap()[first_byte_index] & mask1;
        let mut two_bytes = (existing as u16) << 8;
        let mask2 = if nbits == 8 {
            0xFFu8
        } else {
            !(0xFFu8 << nbits)
        };
        let shift = (8 + (8 - first_byte_cur_bit) - nbits as usize);
        let shifted_input = ((value & mask2) as u16) << shift;
        two_bytes |= shifted_input;
        self.output.as_mut().unwrap()[first_byte_index] = (two_bytes >> 8) as u8;
        if first_byte_cur_bit + nbits as usize > 8 {
            self.output.as_mut().unwrap()[first_byte_index + 1] = two_bytes as u8;
        }
        self.cur_bit += nbits as usize;
        Ok(self.cur_bit / 8)
    }
}
const ILLEGALS: [u8; 6] = [
    0,  
    10, 
    13, 
    34, 
    38, 
    92, 
];
fn is_illegal(val: u8) -> bool {
    ILLEGALS.contains(&val)
}
fn get_illegal_index(val: u8) -> u8 {
    for (i, &illegal) in ILLEGALS.iter().enumerate() {
        if illegal == val {
            return i as u8;
        }
    }
    panic!("unreachable: value is not illegal");
}
pub fn encode(input: &[u8]) -> Result<Vec<u8>, Base122Error> {
    let mut out_written = 0;
    encode_internal(input, None, &mut out_written)?;
    let mut output = vec![0u8; out_written];
    out_written = 0;
    encode_internal(input, Some(&mut output), &mut out_written)?;
    output.truncate(out_written);
    Ok(output)
}
fn encode_internal(input: &[u8], mut output: Option<&mut [u8]>, out_written: &mut usize) -> Result<(), Base122Error> {
    let mut reader = BitReader::new(input);
    let mut out_index = 0;
    let count_only = output.is_none();
    *out_written = 0;
    loop {
        let (nbits, bits) = reader.read(7);
        if nbits == 0 {
            break;
        }
        let bits = if nbits < 7 {
            bits << (7 - nbits)
        } else {
            bits
        };
        if is_illegal(bits) {
            let illegal_index = get_illegal_index(bits);
            let (next_nbits, next_bits) = reader.read(7);
            let next_bits = if next_nbits < 7 && next_nbits > 0 {
                next_bits << (7 - next_nbits)
            } else {
                next_bits
            };
            let mut b1 = 0xC2u8;
            let b2: u8;
            let actual_next_bits: u8;
            if next_nbits == 0 {
                b1 |= 0x7 << 2;
                actual_next_bits = bits;
                b2 = 0x80 | (actual_next_bits & 0x3F);
            } else {
                b1 |= illegal_index << 2;
                actual_next_bits = next_bits;
                b2 = 0x80 | (actual_next_bits & 0x3F);
            }
            let first_bit = (actual_next_bits >> 6) & 1;
            let b1 = b1 | first_bit;
            if count_only {
                *out_written += 1;
            } else {
                if out_index >= output.as_ref().unwrap().len() {
                    return Err(Base122Error::new("output does not have sufficient size"));
                }
                output.as_mut().unwrap()[out_index] = b1;
                out_index += 1;
                *out_written += 1;
            }
            if count_only {
                *out_written += 1;
            } else {
                if out_index >= output.as_ref().unwrap().len() {
                    return Err(Base122Error::new("output does not have sufficient size"));
                }
                output.as_mut().unwrap()[out_index] = b2;
                out_index += 1;
                *out_written += 1;
            }
        } else {
            if count_only {
                *out_written += 1;
            } else {
                if out_index >= output.as_ref().unwrap().len() {
                    return Err(Base122Error::new("output does not have sufficient size"));
                }
                output.as_mut().unwrap()[out_index] = bits;
                out_index += 1;
                *out_written += 1;
            }
        }
    }
    Ok(())
}
fn write_last_7(writer: &mut BitWriter, byte_val: u8, error: &mut Base122Error) -> Result<(), Base122Error> {
    let nbits = 8 - (writer.cur_bit % 8);
    if nbits == 8 {
        *error = Base122Error::new("Decoded data is not a byte multiple");
        return Err(Base122Error::new("Decoded data is not a byte multiple"));
    }
    let mask = !(0xFFu8 << (7 - nbits));
    if (byte_val & mask) != 0 {
        *error = Base122Error::new("Encoded data is malformed. Last byte has extra data.");
        return Err(Base122Error::new("Encoded data is malformed. Last byte has extra data."));
    }
    let shifted_val = byte_val >> (7 - nbits);
    match writer.write(nbits as u8, shifted_val) {
        Ok(_) => Ok(()),
        Err(e) => {
            *error = e.clone();
            Err(e)
        }
    }
}
pub fn decode(input: &[u8]) -> Result<Vec<u8>, Base122Error> {
    let mut writer = BitWriter::new(None, 0);
    let mut out_written = 0;
    decode_internal(input, &mut writer, &mut out_written)?;
    let mut output = vec![0u8; out_written];
    out_written = 0;
    let output_len = output.len();
    let mut writer = BitWriter::new(Some(&mut output), output_len);
    decode_internal(input, &mut writer, &mut out_written)?;
    output.truncate(out_written);
    Ok(output)
}
fn decode_internal(input: &[u8], writer: &mut BitWriter, out_written: &mut usize) -> Result<(), Base122Error> {
    let mut cur_byte = 0;
    let mut error = Base122Error::new("");
    while cur_byte < input.len() {
        if input[cur_byte] >> 7 == 0 {
            let cur_byte_val = input[cur_byte];
            if cur_byte + 1 == input.len() {
                if let Err(e) = write_last_7(writer, cur_byte_val, &mut error) {
                    return Err(e);
                }
            } else {
                if let Err(_) = writer.write(7, cur_byte_val) {
                    return Err(Base122Error::new("Output does not have sufficient size"));
                }
            }
            cur_byte += 1;
        } else {
            let cur_byte_val = input[cur_byte];
            if (cur_byte_val & 0xE2) != 0xC2 {
                return Err(Base122Error::new("First byte of two byte sequence malformed"));
            }
            if cur_byte + 1 >= input.len() {
                return Err(Base122Error::new("Two byte sequence is missing second byte"));
            }
            let next_byte_val = input[cur_byte + 1];
            if (next_byte_val & 0xC0) != 0x80 {
                return Err(Base122Error::new("Second byte of two byte sequence malformed"));
            }
            let illegal_index = (cur_byte_val & 0x1C) >> 2;
            if illegal_index == 0x7 {
                if cur_byte + 2 != input.len() {
                    return Err(Base122Error::new("Got unexpected extra data after shortened two byte sequence"));
                }
                let last_byte_val = ((cur_byte_val as u16) << 6) | ((next_byte_val & 0x3F) as u16);
                if let Err(e) = write_last_7(writer, last_byte_val as u8, &mut error) {
                    return Err(e);
                }
                cur_byte += 2;
            } else if (illegal_index as usize) < ILLEGALS.len() {
                if let Err(_) = writer.write(7, ILLEGALS[illegal_index as usize]) {
                    return Err(Base122Error::new("Output does not have sufficient size"));
                }
                let second_byte_val = ((cur_byte_val as u16) << 6) | ((next_byte_val & 0x3F) as u16);
                if cur_byte + 2 == input.len() {
                    if let Err(e) = write_last_7(writer, second_byte_val as u8, &mut error) {
                        return Err(e);
                    }
                } else {
                    if let Err(_) = writer.write(7, second_byte_val as u8) {
                        return Err(Base122Error::new("Output does not have sufficient size"));
                    }
                }
                cur_byte += 2;
            } else {
                return Err(Base122Error::new("Got unrecognized illegal index"));
            }
        }
    }
    *out_written = writer.cur_bit / 8;
    Ok(())
}
