{{hash.rs}}
```rust
pub type Hash = u32;
pub type Key = &'static str;
pub const EMPTY_KEY: Option<Key> = None;
pub type Data = &'static str;
pub const EMPTY_DATA: Option<Data> = None;
pub const POWER: Hash = 131;
pub const ALTERNATIVE_POWER: Hash = 171;
pub const REHASHER: Hash = 718841;

/// Computes a hash using the given power value.
pub fn hash_by_power(key: Key, power: Hash) -> Hash {
    let mut res: Hash = 0;
    for &byte in key.as_bytes() {
        res = res.wrapping_mul(power).wrapping_add(byte as Hash);
    }
    res
}

/// Computes the default hash using `POWER`.
pub fn hash(key: Key) -> Hash {
    hash_by_power(key, POWER)
}

/// Computes an alternative hash using `ALTERNATIVE_POWER`.
pub fn alternative_hash(key: Key) -> Hash {
    hash_by_power(key, ALTERNATIVE_POWER)
}

/// Compares two keys lexicographically.
pub fn compare_keys(k1: Key, k2: Key) -> std::cmp::Ordering {
    k1.cmp(k2)
}

/// Rehashes an existing hash value.
pub fn rehash(h: Hash) -> Hash {
    h.wrapping_add(REHASHER)
}

```rust
use std::collections::HashMap;
use std::sync::{Arc, RwLock};
use std::fmt;
use crate::check;

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum LogType {
    Log = 0,
    UnitTest = 1,
    OpenHash = 2,
    CuckooHash = 3,
}

pub struct LogInfo {
    priority: u8,
    prefix: &'static str,
    suffix: &'static str,
}

pub struct Logger {
    log_info_table: RwLock<HashMap<LogType, LogInfo>>,
}

impl Logger {
    pub fn new() -> Arc<Self> {
        let mut table = HashMap::new();
        table.insert(LogType::Log, LogInfo { priority: 0, prefix: "log log:\t", suffix: "\n" });
        table.insert(LogType::UnitTest, LogInfo { priority: 0, prefix: "unit testing:\t", suffix: "\n" });
        table.insert(LogType::OpenHash, LogInfo { priority: 0, prefix: "open hash table:\t", suffix: "\n" });
        table.insert(LogType::CuckooHash, LogInfo { priority: 0, prefix: "cuckoo hash table:\t", suffix: "\n" });
        
        Arc::new(Logger {
            log_info_table: RwLock::new(table),
        })
    }
    
    pub fn insert_log(&self, log_type: LogType, priority: u8, format: fmt::Arguments) {
        check!(log_type as usize <= LogType::CuckooHash as usize && priority <= 10);
        let table = self.log_info_table.read().unwrap();
        if let Some(info) = table.get(&log_type) {
            if info.priority <= priority {
                print!("{}{}{}", info.prefix, format, info.suffix);
            }
        }
    }
    
    pub fn change_log_priority(&self, log_type: LogType, new_priority: u8) -> u8 {
        let mut table = self.log_info_table.write().unwrap();
        let old = table.get(&log_type).map(|i| i.priority).unwrap_or(0);
        if let Some(info) = table.get_mut(&log_type) {
            info.priority = new_priority;
        }
        old
    }
}

{{openhash.rs}}
```rust
use std::sync::{Arc, RwLock};
use crate::check;
use crate::hash::{hash, rehash, EMPTY_KEY, compare_keys};

const LOAD_FACTOR: f32 = 0.6;
#[cfg(debug_assertions)]
const EPS: f32 = 1e-3;

#[derive(Clone)]
pub struct OpenEntry {
    key: Option<&'static str>,
    data: Option<&'static str>,
}

pub struct OpenHashTable {
    cur_size: usize,
    max_size: usize,
    arr: Vec<OpenEntry>,
}

impl OpenHashTable {
    pub fn new(initial_size: usize) -> Arc<RwLock<Self>> {
        check!(initial_size != 0);
        check!(initial_size != crate::hash::POWER as usize && initial_size != crate::hash::REHASHER as usize);
        
        let mut arr = Vec::with_capacity(initial_size);
        for _ in 0..initial_size {
            arr.push(OpenEntry { key: None, data: None });
        }
        
        Arc::new(RwLock::new(OpenHashTable {
            cur_size: 0,
            max_size: initial_size,
            arr,
        }))
    }
    
    fn query(&self, key: &str) -> usize {
        check!(Some(key) != EMPTY_KEY);
        let mut h = hash(key);
        loop {
            let idx = (h as usize) % self.max_size;
            let entry = &self.arr[idx];
            if entry.key.is_none() || entry.key == Some(key) {
                return idx;
            }
            h = rehash(h);
        }
    }
    
    pub fn insert(&mut self, key: &'static str, data: &'static str) {
        check!((self.cur_size as f32 / self.max_size as f32) < LOAD_FACTOR + EPS);
        
        self.cur_size += 1;
        
        if (self.cur_size as f32 / self.max_size as f32) > LOAD_FACTOR {
            self.max_size *= 2;
            let old_arr = std::mem::replace(&mut self.arr, Vec::with_capacity(self.max_size));
            
            for _ in 0..self.max_size {
                self.arr.push(OpenEntry { key: None, data: None });
            }
            
            for entry in old_arr {
                if let Some(k) = entry.key {
                    let idx = self.query(k);
                    self.arr[idx] = entry;
                }
            }
        }
        
        let idx = self.query(key);
        check!(self.arr[idx].key.is_none());
        
        self.arr[idx] = OpenEntry { key: Some(key), data: Some(data) };
    }
    
    pub fn find(&self, key: &str) -> Option<&'static str> {
        let idx = self.query(key);
        check!(self.arr[idx].key.is_some());
        self.arr[idx].data
    }
    
    pub fn free_open_hash_table(&mut self) {
        self.arr.clear();
        self.cur_size = 0;
        self.max_size = 0;
    }
}

{{cuckoohash.rs}}
```rust
use std::sync::{Arc, RwLock};
use crate::check;
use crate::hash::{hash, EMPTY_KEY, compare_keys, POWER, ALTERNATIVE_POWER};

const LOAD_FACTOR: f32 = 1.0;
#[cfg(debug_assertions)]
const EPS: f32 = 1e-3;

#[derive(Clone)]
pub struct CuckooEntry {
    key: Option<&'static str>,
    data: Option<&'static str>,
    marker: u32,
}

pub struct CuckooHashTable {
    cur_size: u32,
    cur_marker: u32,
    max_size: u32,
    first_arr: Vec<CuckooEntry>,
    second_arr: Vec<CuckooEntry>,
}

impl CuckooHashTable {
    pub fn new(initial_size: usize) -> Arc<RwLock<Self>> {
        check!(initial_size >= 2);
        check!(initial_size != POWER as usize && initial_size != ALTERNATIVE_POWER as usize);
        
        let size = initial_size / 2;
        let mut first_arr = Vec::with_capacity(size);
        let mut second_arr = Vec::with_capacity(size);
        
        for _ in 0..size {
            first_arr.push(CuckooEntry { key: None, data: None, marker: 0 });
            second_arr.push(CuckooEntry { key: None, data: None, marker: 0 });
        }
        
        Arc::new(RwLock::new(CuckooHashTable {
            cur_size: 0,
            cur_marker: 0,
            max_size: size as u32,
            first_arr,
            second_arr,
        }))
    }
    
    pub fn insert(&mut self, key: &'static str, data: &'static str) {
        self.cur_marker += 1;
        
        if 1.0 + self.cur_size as f32 > self.max_size as f32 * LOAD_FACTOR {
            self.resize();
        }
        
        self.cur_size += 1;
        
        let first_idx = self.get_first_entry_index(key);
        {
            let entry = &mut self.first_arr[first_idx];
            if self.try_to_store(key, data, entry) {
                return;
            }
        }
        
        let second_idx = self.get_second_entry_index(key);
        {
            let entry = &mut self.second_arr[second_idx];
            if self.try_to_store(key, data, entry) {
                return;
            }
        }
        
        let mut current_key = key;
        let mut current_data = data;
        let mut first_idx = first_idx;
        
        loop {
            {
                let entry = &mut self.first_arr[first_idx];
                self.swap_key_data_entry(&mut current_key, &mut current_data, entry);
                entry.marker = self.cur_marker;
            }
            
            let second_idx = self.get_second_entry_index(current_key);
            {
                let entry2 = &self.second_arr[second_idx];
                if entry2.marker == self.cur_marker {
                    drop(entry2);
                    self.refill();
                    self.insert(current_key, current_data);
                    return;
                }
            }
            
            {
                let entry2 = &mut self.second_arr[second_idx];
                if self.try_to_store(current_key, current_data, entry2) {
                    return;
                }
                self.swap_key_data_entry(&mut current_key, &mut current_data, entry2);
                entry2.marker = self.cur_marker;
            }
            
            first_idx = self.get_first_entry_index(current_key);
            {
                let entry1 = &self.first_arr[first_idx];
                if entry1.marker == self.cur_marker {
                    drop(entry1);
                    self.refill();
                    self.insert(current_key, current_data);
                    return;
                }
            }
            
            {
                let entry1 = &mut self.first_arr[first_idx];
                if self.try_to_store(current_key, current_data, entry1) {
                    return;
                }
            }
        }
    }
    
    pub fn find(&self, key: &str) -> Option<&'static str> {
        let first_idx = self.get_first_entry_index(key);
        let entry = &self.first_arr[first_idx];
        if entry.key.is_some() && entry.key == Some(key) {
            return entry.data;
        }
        
        let second_idx = self.get_second_entry_index(key);
        let entry2 = &self.second_arr[second_idx];
        check!(entry2.key.is_some() && entry2.key == Some(key));
        entry2.data
    }
    
    fn resize(&mut self) {
        self.recreate(self.max_size * 2);
    }
    
    fn refill(&mut self) {
        self.recreate(self.max_size + 1);
    }
    
    fn recreate(&mut self, new_size: u32) {
        check!(new_size > 0);
        check!(new_size as f32 * LOAD_FACTOR + EPS > self.cur_size as f32);
        
        let old_first = std::mem::replace(&mut self.first_arr, Vec::with_capacity(new_size as usize));
        let old_second = std::mem::replace(&mut self.second_arr, Vec::with_capacity(new_size as usize));
        
        self.cur_size = 0;
        self.cur_marker = 0;
        let old_max_size = self.max_size;
        self.max_size = new_size;
        
        for _ in 0..new_size {
            self.first_arr.push(CuckooEntry { key: None, data: None, marker: 0 });
            self.second_arr.push(CuckooEntry { key: None, data: None, marker: 0 });
        }
        
        for entry in old_first {
            if let (Some(k), Some(d)) = (entry.key, entry.data) {
                self.insert(k, d);
            }
        }
        
        for entry in old_second {
            if let (Some(k), Some(d)) = (entry.key, entry.data) {
                self.insert(k, d);
            }
        }
    }
    
    fn get_first_entry_index(&self, key: &str) -> usize {
        let h = hash(key);
        (h % self.max_size) as usize
    }
    
    fn get_second_entry_index(&self, key: &str) -> usize {
        let h = hash(key);
        (h % self.max_size) as usize
    }
    
    fn get_first_entry(&mut self, key: &str) -> &mut CuckooEntry {
        let idx = self.get_first_entry_index(key);
        &mut self.first_arr[idx]
    }
    
    fn get_second_entry(&mut self, key: &str) -> &mut CuckooEntry {
        let idx = self.get_second_entry_index(key);
        &mut self.second_arr[idx]
    }
    
    fn try_to_store(&mut self, key: &'static str, data: &'static str, entry: &mut CuckooEntry) -> bool {
        if entry.key.is_none() {
            entry.key = Some(key);
            entry.data = Some(data);
            true
        } else {
            check!(entry.key != Some(key));
            false
        }
    }
