//! LZMA Packet definitions and types.

/// Types of LZMA packets used in the compression algorithm.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub enum LZMAPacketType {
    /// Literal byte packet (single uncompressed byte).
    Literal,
    /// Match packet (backward reference with length and distance).
    Match,
    /// Short repeat packet (short distance repeat).
    ShortRep,
    /// Long repeat packet (long distance repeat).
    LongRep,
}

/// Represents a single LZMA packet.
#[derive(Debug, Clone)]
pub struct LZMAPacket {
    pub packet_type: LZMAPacketType,
    pub length: usize,
    pub distance: u32,
}

impl LZMAPacket {
    /// Creates a new literal packet.
    pub fn literal() -> Self {
        Self {
            packet_type: LZMAPacketType::Literal,
            length: 1,
            distance: 0,
        }
    }

    /// Creates a new match packet.
    pub fn match_packet(length: usize, distance: u32) -> Self {
        Self {
            packet_type: LZMAPacketType::Match,
            length,
            distance,
        }
    }
}

//! Packet slab allocator for efficient memory management.

/// A slab allocator for managing LZMA packets efficiently.
pub struct PacketSlab {
    capacity: usize,
    used: usize,
}

impl PacketSlab {
    /// Creates a new empty packet slab.
    pub fn new() -> Self {
        Self {
            capacity: 1024,
            used: 0,
        }
    }

    /// Creates a new packet slab with the specified capacity.
    pub fn with_capacity(capacity: usize) -> Self {
        Self { capacity, used: 0 }
    }

    /// Returns the capacity of the slab.
    pub fn capacity(&self) -> usize {
        self.capacity
    }

    /// Returns the number of used slots in the slab.
    pub fn len(&self) -> usize {
        self.used
    }

    /// Returns true if the slab is empty.
    pub fn is_empty(&self) -> bool {
        self.used == 0
    }

    /// Clears the slab.
    pub fn clear(&mut self) {
        self.used = 0;
    }
}

impl Default for PacketSlab {
    fn default() -> Self {
        Self::new()
    }
}

/// Represents a single undo operation for the packet slab.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub struct PacketSlabUndo {
    pub index: usize,
    pub operation: UndoOperation,

/// Types of undo operations.

pub enum UndoOperation {
    Insert,
    Delete,
    Update,
}

/// A stack for managing packet slab undo operations.
pub struct PacketSlabUndoStack {
    stack: Vec<PacketSlabUndo>,

impl PacketSlabUndoStack {
    /// Creates a new empty undo stack.

        Self { stack: Vec::new() }
    }

    /// Creates a new undo stack with the specified capacity.
    pub fn with_capacity(capacity: usize) -> Self {
        Self {
            stack: Vec::with_capacity(capacity),
        }
    }

    }

        self.stack.pop()

        self.stack.is_empty()

        self.stack.len()
