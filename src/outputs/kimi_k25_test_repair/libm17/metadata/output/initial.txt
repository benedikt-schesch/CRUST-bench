use crate::math::q_abs_diff;

pub const LSF_SYNC_SYMBOLS: [i8; 8] = [3, 3, 3, 3, -3, -3, 3, -3];
pub const STR_SYNC_SYMBOLS: [i8; 8] = [-3, -3, -3, -3, 3, 3, -3, 3];
pub const PKT_SYNC_SYMBOLS: [i8; 8] = [3, -3, 3, 3, -3, -3, -3, -3];
pub const SYMBOL_LEVELS: [f32; 4] = [-3.0, -1.0, 1.0, 3.0];
pub const NUM_STATES: usize = 1 << (5 - 1);
static mut PREV_METRICS: [u32; NUM_STATES] = [0; NUM_STATES];
static mut CURR_METRICS: [u32; NUM_STATES] = [0; NUM_STATES];
static mut PREV_METRICS_DATA: [u32; NUM_STATES] = [0; NUM_STATES];
static mut CURR_METRICS_DATA: [u32; NUM_STATES] = [0; NUM_STATES];
static mut VITERBI_HISTORY: [u16; 244] = [0; 244];

pub fn viterbi_decode(out: &mut [u8], input: &[u16], len: u16) -> u32 {
    if len > 244 * 2 {
        eprintln!("Input size exceeds max history");
    }
    unsafe {
        viterbi_reset();
    }
    let mut pos = 0usize;
    for i in (0..len as usize).step_by(2) {
        if i + 1 < input.len() {
            let s0 = input[i];
            let s1 = input[i + 1];
            unsafe {
                viterbi_decode_bit(s0, s1, pos);
            }
            pos += 1;
        }
    }
    unsafe { viterbi_chainback(out, pos, len / 2) }
}

pub fn viterbi_decode_punctured(
    out: &mut [u8],
    input: &[u16],
    punct: &[u8],
    in_len: u16,
    p_len: u16,
) -> u32 {
    if in_len > 244 * 2 {
        eprintln!("Input size exceeds max history");
    }
    let mut umsg = [0u16; 244 * 2];
    let mut p: usize = 0;
    let mut u: usize = 0;
    let mut i: usize = 0;
    while i < in_len as usize {
        if punct[p] != 0 {
            umsg[u] = input[i];
            i += 1;
        } else {
            umsg[u] = 0x7FFF;
        }
        u += 1;
        p += 1;
        p %= p_len as usize;
    }
    let cost = unsafe { viterbi_decode(out, &umsg[..u], u as u16) };
    cost.wrapping_sub(((u - in_len as usize) as u32).wrapping_mul(0x7FFF))
}

pub fn viterbi_decode_bit(s0: u16, s1: u16, pos: usize) {
    const COST_TABLE_0: [u16; 8] = [0, 0, 0, 0, 0xFFFF, 0xFFFF, 0xFFFF, 0xFFFF];
    const COST_TABLE_1: [u16; 8] = [0, 0xFFFF, 0xFFFF, 0, 0, 0xFFFF, 0xFFFF, 0];
    
    unsafe {
        for i in 0..NUM_STATES / 2 {
            let metric = q_abs_diff(COST_TABLE_0[i], s0) + q_abs_diff(COST_TABLE_1[i], s1);
            let m0 = PREV_METRICS[i].wrapping_add(metric as u32);
            let m1 = PREV_METRICS[i + NUM_STATES / 2].wrapping_add(0x1FFFE - metric as u32);
            let m2 = PREV_METRICS[i].wrapping_add(0x1FFFE - metric as u32);
            let m3 = PREV_METRICS[i + NUM_STATES / 2].wrapping_add(metric as u32);
            
            let i0 = 2 * i;
            let i1 = i0 + 1;
            
            if m0 >= m1 {
                VITERBI_HISTORY[pos] |= 1 << i0;
                CURR_METRICS[i0] = m1;
            } else {
                VITERBI_HISTORY[pos] &= !(1 << i0);
                CURR_METRICS[i0] = m0;
            }
            
            if m2 >= m3 {
                VITERBI_HISTORY[pos] |= 1 << i1;
                CURR_METRICS[i1] = m3;
            } else {
                VITERBI_HISTORY[pos] &= !(1 << i1);
                CURR_METRICS[i1] = m2;
            }
        }
        
        let mut tmp = [0u32; NUM_STATES];
        for i in 0..NUM_STATES {
            tmp[i] = CURR_METRICS[i];
        }
        for i in 0..NUM_STATES {
            CURR_METRICS[i] = PREV_METRICS[i];
            PREV_METRICS[i] = tmp[i];
        }
    }
}

fn viterbi_chainback(out: &mut [u8], pos: usize, len: u16) -> u32 {
    let mut state: u8 = 0;
    let mut bit_pos: usize = (len as usize) + 4;
    let out_len = ((len as usize - 1) / 8) + 1;
    for i in 0..out_len {
        out[i] = 0;
    }
    
    let mut pos = pos;
    while pos > 0 {
        bit_pos -= 1;
        pos -= 1;
        unsafe {
            let bit = (VITERBI_HISTORY[pos] >> (state >> 4)) & 1;
            state >>= 1;
            if bit != 0 {
                state |= 0x80;
                out[bit_pos / 8] |= 1 << (7 - (bit_pos % 8));
            }
        }
    }
    
    unsafe {
        let mut cost = PREV_METRICS[0];
        for i in 0..NUM_STATES {
            if PREV_METRICS[i] < cost {
                cost = PREV_METRICS[i];
            }
        }
        cost
    }
}

fn viterbi_reset() {
    unsafe {
        for i in 0..244 {
            VITERBI_HISTORY[i] = 0;
        }
        for i in 0..NUM_STATES {
            PREV_METRICS[i] = 0;
            CURR_METRICS[i] = 0;
            PREV_METRICS_DATA[i] = 0;
            CURR_METRICS_DATA[i] = 0;
        }
    }
}

use crate::types::{SYM_PER_FRA, SYM_PER_SWD, SYM_PER_PLD, PreambleType, FrameType, LSF};
use crate::encode::{SYMBOL_MAP, EOT_SYMBOLS, conv_encode_lsf, conv_encode_stream_frame, conv_encode_packet_frame};
use crate::payload::{extract_lich, unpack_lich};
use crate::math::encode_lich;
use crate::phy::{SYNC_LSF, SYNC_STR, SYNC_PKT, reorder_bits, randomize_bits};

pub fn send_preamble(out: &mut [f32], cnt: &mut usize, ptype: PreambleType) {
    if matches!(ptype, PreambleType::Bert) {
        for _ in 0..SYM_PER_FRA / 2 {
            out[*cnt] = -3.0;
            *cnt += 1;
            out[*cnt] = 3.0;
            *cnt += 1;
        }
    } else {
        for _ in 0..SYM_PER_FRA / 2 {
            out[*cnt] = 3.0;
            *cnt += 1;
            out[*cnt] = -3.0;
            *cnt += 1;
        }
    }
}

pub fn send_syncword(out: &mut [f32], cnt: &mut usize, syncword: u16) {
    for i in (0..SYM_PER_SWD * 2).step_by(2) {
        let idx = ((syncword >> (14 - i)) & 3) as usize;
        out[*cnt] = SYMBOL_MAP[idx] as f32;
        *cnt += 1;
    }
}

pub fn send_data(out: &mut [f32], cnt: &mut usize, inp: &[u8]) {
    for i in 0..SYM_PER_PLD {
        let bit1 = inp[2 * i] as usize;
        let bit2 = inp[2 * i + 1] as usize;
        let idx = bit1 * 2 + bit2;
        out[*cnt] = SYMBOL_MAP[idx] as f32;
        *cnt += 1;
    }
}

pub fn send_eot(out: &mut [f32], cnt: &mut usize) {
    for i in 0..SYM_PER_FRA {
        out[*cnt] = EOT_SYMBOLS[i % 8];
        *cnt += 1;
    }
}

pub fn send_frame(out: &mut [f32], data: &[u8], ftype: FrameType, lsf: &LSF, lich_cnt: u8, fn_num: u16) {
    let mut lich = [0u8; 6];
    let mut lich_encoded = [0u8; 12];
    let mut enc_bits = [0u8; SYM_PER_PLD * 2];
    let mut rf_bits = [0u8; SYM_PER_PLD * 2];
    let mut sym_cnt: usize = 0;
    
    match ftype {
        FrameType::Lsf => {
            send_syncword(out, &mut sym_cnt, SYNC_LSF);
            conv_encode_lsf(&mut enc_bits, lsf);
        },
        FrameType::Str => {
            send_syncword(out, &mut sym_cnt, SYNC_STR);
            extract_lich(&mut lich, lich_cnt, lsf);
            lich_encoded = encode_lich(&lich);
            unpack_lich(&mut enc_bits[..96], &lich_encoded);
            conv_encode_stream_frame(&mut enc_bits[96..], data, fn_num);
        },
        FrameType::Pkt => {
            send_syncword(out, &mut sym_cnt, SYNC_PKT);
            conv_encode_packet_frame(&mut enc_bits, data);
        },
    }
    
    reorder_bits(&mut rf_bits, &enc_bits);
    randomize_bits(&mut rf_bits);
    send_data(out, &mut sym_cnt, &rf_bits);
}
