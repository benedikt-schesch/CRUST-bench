fn sane_istest(x: char, mask: u8) -> bool {
    let x = x as usize;
    if x >= 256 {
        return false;
    }
    (SANE_CTYPE[x] & mask) != 0
}

fn isdigit(x: char) -> bool {
    sane_istest(x, GIT_DIGIT)
}

fn isalpha(x: char) -> bool {
    sane_istest(x, GIT_ALPHA)
}

fn isalnum(x: char) -> bool {
    sane_istest(x, GIT_ALPHA | GIT_DIGIT)
}

fn toupper(x: char) -> char {
    if sane_istest(x, GIT_ALPHA) {
        ((x as u8) & !0x20) as char
    } else {
        x
    }
}

fn time_t_to_tm(mut t: i64) -> Atm {
    let mut tm = Atm {
        tm_sec: 0,
        tm_min: 0,
        tm_hour: 0,
        tm_mday: 0,
        tm_mon: 0,
        tm_year: 0,
        tm_wday: 0,
        tm_yday: 0,
        tm_isdst: 0,
        tm_usec: 0,
    };

    if t < 0 {
        t = 0;
    }

    tm.tm_sec = (t % 60) as i32;
    t /= 60;
    tm.tm_min = (t % 60) as i32;
    t /= 60;
    tm.tm_hour = (t % 24) as i32;
    t /= 24;

    let days_since_epoch = t;
    tm.tm_wday = ((days_since_epoch + 4) % 7) as i32;

    let mut year = 1970;
    let mut days = days_since_epoch;
    loop {
        let year_days = if is_leap_year(year) { 366 } else { 365 };
        if days < year_days {
            break;
        }
        days -= year_days;
        year += 1;
    }
    tm.tm_year = year - 1900;
    tm.tm_yday = days as i32;

    let mut mon = 0;
    while mon < 11 {
        let dim = days_in_month(year, mon);
        if days < dim {
            break;
        }
        days -= dim;
        mon += 1;
    }
    tm.tm_mon = mon;
    tm.tm_mday = (days + 1) as i32;

    tm
}

pub fn tm_to_time_t(tm: &Atm) -> Option<i64> {
    const MDAYS: [i32; 12] = [0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334];
    
    let year = tm.tm_year - 70;
    let month = tm.tm_mon;
    let mut day = tm.tm_mday;
    
    if year < 0 || year > 129 {
        return None;
    }
    if month < 0 || month > 11 {
        return None;
    }
    
    if month < 2 || (year + 2) % 4 != 0 {
        day -= 1;
    }
    
    if tm.tm_hour < 0 || tm.tm_min < 0 || tm.tm_sec < 0 {
        return None;
    }
    
    let days_since_epoch = year * 365 + (year + 1) / 4 + MDAYS[month as usize] + day;
    let seconds = days_since_epoch as i64 * 24 * 60 * 60 
        + tm.tm_hour as i64 * 60 * 60 
        + tm.tm_min as i64 * 60 
        + tm.tm_sec as i64;
    
    Some(seconds)
}

pub fn match_string(date: &str, format: &str) -> i32 {
    let mut i = 0;
    let mut d_chars = date.chars();
    let mut f_chars = format.chars();
    
    loop {
        let d = d_chars.next();
        let f = f_chars.next();
        
        match (d, f) {
            (Some(dc), Some(fc)) => {
                if dc == fc || toupper(dc) == toupper(fc) {
                    i += 1;
                    continue;
                }
                if !isalnum(dc) {
                    break;
                }
                return 0;
            }
            (Some(dc), None) => {
                if !isalnum(dc) {
                    break;
                }
                return 0;
            }
            (None, _) => break,
        }
    }
    i
}

pub fn skip_alpha(date: &str) -> i32 {

pub fn match_alpha(date: &str, tm: &mut Atm, offset: &mut i32) -> i32 {

            return match_len;
        }
    }
    
    for tz in &TIMEZONE_NAMES {
        let match_len = match_string(date, tz.name);
        let tz_name_len = tz.name.len() as i32;
        if match_len >= 3 || match_len == tz_name_len {
            let mut off = tz.offset;
            if tz.dst {
                off += 1;
            }
            if *offset == -1 {
                *offset = 60 * off;
            }
            return match_len;
        }
    }
    
    if match_string(date, "PM") == 2 {
        tm.tm_hour = (tm.tm_hour % 12) + 12;
        return 2;
    }
    if match_string(date, "AM") == 2 {
        tm.tm_hour = (tm.tm_hour % 12) + 0;
        return 2;
    }
    
    skip_alpha(date)

pub fn is_date(year: i32, month: i32, day: i32, now_tm: &Atm, _now: TimeT, tm: &mut Atm) -> i32 {
    if month > 0 && month < 13 && day > 0 && day < 32 {
        let mut check = *tm;
        let r: &mut Atm;
        let use_now = now_tm.tm_year >= 0;
        
        if use_now {
            r = &mut check;
        } else {
            r = tm;
        }
        
        r.tm_mon = month - 1;
        r.tm_mday = day;
        
        if year == -1 {
            if !use_now {
                return 1;
            }
            r.tm_year = now_tm.tm_year;
        } else if year >= 1970 && year < 2100 {
            r.tm_year = year - 1900;
        } else if year > 70 && year < 100 {
            r.tm_year = year;
        } else if year < 38 {
            r.tm_year = year + 100;
        } else {
            return 0;
        }
        
        if !use_now {
            return 1;
        }
        
        tm.tm_mon = r.tm_mon;
        tm.tm_mday = r.tm_mday;
        if year != -1 {
            tm.tm_year = r.tm_year;
        }
        return 1;
    }
    0
}

fn parse_long(s: &str) -> (i64, usize) {
    let mut i = 0;
    let bytes = s.as_bytes();
    if bytes.is_empty() {
        return (0, 0);
    }
    
    let mut sign = 1;
    if bytes[0] == b'-' {
        sign = -1;
        i = 1;
    } else if bytes[0] == b'+' {
        i = 1;
    }
    
    let start = i;
    let mut val: i64 = 0;
    while i < bytes.len() && bytes[i].is_ascii_digit() {
        val = val * 10 + (bytes[i] - b'0') as i64;
        i += 1;
    }
    
    if i == start {
        return (0, 0);
    }
    (val * sign, i)
}

    }

pub fn match_digit(date: &str, tm: &mut Atm, offset: &mut i32, tm_gmt: &mut i32) -> i32 {

    if len == 0 {
        return 0;
    }
    
    if num >= 100000000 && nodate(tm) == 1 {
        let time = num as i64;
        *tm = time_t_to_tm(time);
        *tm_gmt = 1;
        return len as i32;
    }
    
    if len < date.len() {
        let c = date.chars().nth(len).unwrap();
        if c == ':' || c == '.' || c == '/' || c == '-' {
            if len + 1 < date.len() {
                let next_c = date.chars().nth(len + 1).unwrap();
                if isdigit(next_c) {
                    let end_str = &date[len..];
                    let match_len = match_multi_number(num, c, date, end_str, tm, 0);
                    if match_len > 0 {
                        return match_len;
                    }
                }
            }
        }
    }
    
    let mut n = 0;
    let bytes = date.as_bytes();
    while n < bytes.len() && isdigit(bytes[n] as char) {
        n += 1;
    }
    
    if n == 4 {
        if num <= 1400 && *offset == -1 {
            let minutes = (num % 100) as i32;
            let hours = (num / 100) as i32;
            *offset = hours * 60 + minutes;
        } else if num > 1900 && num < 2100 {
            tm.tm_year = (num - 1900) as i32;
        }
        return n as i32;
    }
    
    if n > 2 {
        return n as i32;
    }
    
    if num > 0 && num < 32 && tm.tm_mday < 0 {
        tm.tm_mday = num as i32;
        return n as i32;
    }
    
    if n == 2 && tm.tm_year < 0 {
        if num < 10 && tm.tm_mday >= 0 {
            tm.tm_year = (num + 100) as i32;
            return n as i32;
        }
        if num >= 70 {
            tm.tm_year = num as i32;
            return n as i32;
        }
    }
    
    if num > 0 && num < 13 && tm.tm_mon < 0 {
        tm.tm_mon = (num - 1) as i32;
    }
    
    n as i32
}
