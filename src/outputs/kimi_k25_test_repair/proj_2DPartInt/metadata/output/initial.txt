pub fn fill_grid(
    num_particles: usize,
    x_squares: i32,
    y_squares: i32,
    square_length: f64,
    particles: &mut [data::Particle],
    grid: &mut Vec<Option<&mut data::Particle>>,
    grid_lasts: &mut Vec<Option<&mut data::Particle>>,
) {
    let total_squares = (x_squares * y_squares) as usize;
    grid.clear();
    grid_lasts.clear();
    grid.resize(total_squares, None);
    grid_lasts.resize(total_squares, None);
    
    for i in 0..num_particles {
        particles[i].next = None;
        let square_ind = find_square(
            particles[i].x_coordinate,
            particles[i].y_coordinate,
            x_squares,
            y_squares,
            square_length,
        );
        
        if square_ind < 0 {
            continue;
        }
        
        let square_ind = square_ind as usize;
        
        if grid_lasts[square_ind].is_none() {
            grid[square_ind] = Some(&mut particles[i]);
            grid_lasts[square_ind] = Some(&mut particles[i]);
        } else {
            if let Some(ref mut last) = grid_lasts[square_ind] {
                let next_ptr = &mut particles[i] as *mut data::Particle;
                last.next = Some(Box::new(data::Particle {
                    x_coordinate: particles[i].x_coordinate,
                    y_coordinate: particles[i].y_coordinate,
                    radius: particles[i].radius,
                    next: None,
                    idx: particles[i].idx,
                }));
            }
            grid_lasts[square_ind] = Some(&mut particles[i]);
        }
    }
}

pub fn compute_contacts(
    grid: &Vec<Option<&data::Particle>>,
    x_squares: i32,
    y_squares: i32,
    square_length: f64,
    contacts: &mut [data::Contact],
) -> usize {
    let mut k = 0;
    
    for row in 0..y_squares {
        for col in 0..x_squares {
            let square_idx = (row * x_squares + col) as usize;
            
            if square_idx >= grid.len() || grid[square_idx].is_none() {
                continue;
            }
            
            let first = grid[square_idx].unwrap();
            let mut p = Some(first);
            
            while let Some(p_particle) = p {
                let mut other = Some(first);
                while let Some(other_particle) = other {
                    if other_particle as *const _ != p_particle as *const _ {
                        let overlap = functions::compute_overlap(p_particle, other_particle);
                        if overlap > 0.0 {
                            contacts[k].p1_idx = p_particle.idx as usize;
                            contacts[k].p2_idx = other_particle.idx as usize;
                            contacts[k].overlap = overlap;
                            k += 1;
                        }
                    }
                    other = other_particle.next.as_ref().map(|b| &**b);
                }
                
                let left_col = find_col(p_particle.x_coordinate - p_particle.radius * 2.0, x_squares, square_length);
                let left_col = if left_col == -2 { 0 } else { left_col };
                let right_col = find_col(p_particle.x_coordinate + p_particle.radius * 2.0, x_squares, square_length);
                let right_col = if right_col == -1 { x_squares - 1 } else { right_col };
                
                let bottom_row = find_row(p_particle.y_coordinate - p_particle.radius * 2.0, y_squares, square_length);
                let bottom_row = if bottom_row == -2 { 0 } else { bottom_row };
                let top_row = find_row(p_particle.y_coordinate + p_particle.radius * 2.0, y_squares, square_length);
                let top_row = if top_row == -1 { y_squares - 1 } else { top_row };
                
                for neighbor_row in bottom_row..=top_row {
                    for neighbor_col in left_col..=right_col {
                        let neighbor_square_idx = (neighbor_row * x_squares + neighbor_col) as usize;
                        
                        if neighbor_square_idx == square_idx || neighbor_square_idx >= grid.len() {
                            continue;
                        }
                        
                        if let Some(mut other) = grid[neighbor_square_idx] {
                            while let Some(other_particle) = Some(other) {
                                let overlap = functions::compute_overlap(p_particle, other_particle);
                                if overlap > 0.0 {
                                    contacts[k].p1_idx = p_particle.idx as usize;
                                    contacts[k].p2_idx = other_particle.idx as usize;
                                    contacts[k].overlap = overlap;
                                    k += 1;
                                }
                                other = match other_particle.next.as_ref() {
                                    Some(b) => &**b,
                                    None => break,
                                };
                            }
                        }
                    }
                }
                
                p = p_particle.next.as_ref().map(|b| &**b);
            }
        }
    }
    
    k
}
