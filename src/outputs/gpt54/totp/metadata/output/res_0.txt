pub const SIZE_MAX: usize = !0;

pub fn memset(s: &mut [u8], c: i32, n: usize) -> &mut [u8] {
    let limit = n.min(s.len());
    let byte = c as u8;
    let mut i = 0usize;
    while i < limit {
        s[i] = byte;
        i += 1;
    }
    s
}

pub fn memcpy<'a>(dst: &'a mut [u8], src: &'a [u8], n: usize) -> &'a mut [u8] {
    let limit = n.min(dst.len()).min(src.len());
    let mut i = 0usize;
    while i < limit {
        dst[i] = src[i];
        i += 1;
    }
    dst
}

pub fn strlen(s: &str) -> usize {
    s.len()
}

pub fn from_base32(s: &str, buf: &mut [u8], cap: usize) -> usize {

        return 0;
    }
    if cap < ((std::strlen(s) + 1) / 8) * 5
        || cap > buf.len() && buf.len() < ((std::strlen(s) + 1) / 8) * 5
    {
        return 0;
    }

    let bytes = s.as_bytes();
    let mut i = 0usize;

    while i * 8 < bytes.len() {
        let mut v = [0u8; 8];
        let mut j = 0usize;
        while j < 8 {
            let c = bytes[i * 8 + j];
            v[j] = if c == b'=' {
                0
            } else if c >= b'A' && c <= b'Z' {
                c - b'A'
            } else if c >= b'a' && c <= b'z' {
                c - b'a'
            } else if c >= b'2' && c <= b'7' {
                c - b'2' + 26
            } else {
                return 0;
            };
            j += 1;
        }

        let base = i * 5;
        if base + 4 >= buf.len() {
            return 0;
        }

        buf[base] = (v[0] << 3) | (v[1] >> 2);
        buf[base + 1] = (v[1] << 6) | (v[2] << 1) | (v[3] >> 4);
        buf[base + 2] = (v[3] << 4) | (v[4] >> 1);
        buf[base + 3] = (v[4] << 7) | (v[5] << 2) | (v[6] >> 3);

pub fn unpack64(x: u64, a: &mut [u8; 8]) {
    let hi = (x >> 32) as u32;
    let lo = x as u32;

    let mut hi_bytes = [0u8; 4];
    let mut lo_bytes = [0u8; 4];
    unpack32(hi, &mut hi_bytes);
    unpack32(lo, &mut lo_bytes);

    a[0] = hi_bytes[0];
    a[1] = hi_bytes[1];
    a[2] = hi_bytes[2];
    a[3] = hi_bytes[3];
    a[4] = lo_bytes[0];
    a[5] = lo_bytes[1];
    a[6] = lo_bytes[2];
    a[7] = lo_bytes[3];
}

pub fn pack32(a: &[u8; 4]) -> u32 {
    ((a[0] as u32) << 24) | ((a[1] as u32) << 16) | ((a[2] as u32) << 8) | (a[3] as u32)
}

pub fn unpack32(x: u32, a: &mut [u8; 4]) {
    a[0] = (x >> 24) as u8;
    a[1] = (x >> 16) as u8;
    a[2] = (x >> 8) as u8;
    a[3] = x as u8;
}

    let s = n as u32;
    (x << s) | (x >> (32 - s))
