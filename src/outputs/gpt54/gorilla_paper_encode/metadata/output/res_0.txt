        i += 1;

pub struct BitWriter {
    pub cache: [U8; 1024],
    pub pos: U32,
    pub byte: U8,
    pub bit_count: U8,
}

impl BitWriter {
    pub fn bitwriter_init(&mut self) -> i32 {
        self.cache = [0; 1024];
        self.pos = 0;
        self.byte = 0;
        self.bit_count = 8;
        0
    }

    pub fn append_to_cache(&mut self) -> i32 {
        if self.pos as usize >= self.cache.len() {
            return -1;
        }
        self.cache[self.pos as usize] = self.byte;
        self.pos += 1;
        0
    }

    pub fn write_bit(&mut self, bit: bool) -> i32 {
        if bit {
            self.byte |= 1u8 << (self.bit_count - 1);
        }
        self.bit_count -= 1;
        if self.bit_count == 0 {
            if self.append_to_cache() != 0 {
                return -1;
            }
            self.byte = 0;
            self.bit_count = 8;
        }
        0
    }

    pub fn write_byte(&mut self, b: U8) -> i32 {
        self.byte |= b >> (8 - self.bit_count);
        if self.append_to_cache() != 0 {
            return -1;
        }
        self.byte = b << self.bit_count;
        0
    }

    pub fn write_bits(&mut self, mut u: U64, mut nbits: i32) -> i32 {
        if !(0..=64).contains(&nbits) {
            return -1;
        }
        if nbits == 0 {
            return 0;
        }
        let mut byte: U8;
        u <<= 64 - nbits as u32;
        while nbits >= 8 {
            byte = (u >> 56) as U8;
            if self.write_byte(byte) != 0 {
                return -1;
            }
            u <<= 8;
            nbits -= 8;
        }
        while nbits > 0 {
            if self.write_bit((u >> 63) != 0) != 0 {
                return -2;
            }
            u <<= 1;
            nbits -= 1;
        }
        0
    }

    pub fn write_flush(&mut self, bit: bool) -> i32 {
        while self.bit_count != 8 {
            if self.write_bit(bit) != 0 {
                return -1;
            }
        }
        0
    }
}

pub struct BitReader<'a> {
    pub data: &'a [u8],
    pub len: U32,
    pub v: U64,
    pub n: U32,
}

impl BitReader<'_> {
    pub fn bit_readbuf(&mut self) -> i32 {
        let available = self.data.len().min(self.len as usize);
        if available == 0 {
            self.v = 0;
            self.n = 0;
            return 0;
        }

        let byte_n = available.min(8);
        self.v = 0;
        self.n = 0;

        for i in 0..byte_n {
            self.n += 8;
            self.v |= (self.data[i] as U64) << (64 - self.n);
        }

        self.data = &self.data[byte_n..];
        self.len -= byte_n as U32;
        0
    }

    pub fn bitread_reset(&mut self, _data: &mut [u8]) -> i32 {
        self.data = &[];
        self.len = 0;
        self.n = 0;
        self.v = 0;
        0
    }

    pub fn read_bits(&mut self, nbits: U32) -> U64 {
        if nbits > 64 {
            return !0u64;
        }

        let mut remaining = nbits;
        let mut out: U64 = 0;

        while remaining > 0 {
            if self.n == 0 {
                self.bit_readbuf();
                if self.n == 0 {
                    return !0u64;
                }
            }

            let take = remaining.min(self.n);
            let chunk = self.v >> (64 - take);
            out = (out << take) | chunk;
            self.v <<= take;
            self.n -= take;
            remaining -= take;
        }

        out
    }

    pub fn read_bit(&mut self) -> U64 {
        self.read_bits(1)
    }

    pub fn can_read_bitfast(&mut self) -> bool {
        self.n > 1
    }

    pub fn read_bitfast(&mut self) -> bool {
        let v = (self.v & (1u64 << 63)) != 0;
        self.v <<= 1;
        if self.n > 0 {
            self.n -= 1;
        }
        v
    }

    pub fn read_next(&mut self) -> bool {
        false
    }
}

pub fn bitslen(mut x: U64) -> i32 {
    let mut n = 0;
    if x >= (1u64 << 32) {
        x >>= 32;
        n = 32;
    }
    if x >= (1u64 << 16) {
        x >>= 16;
        n += 16;
    }
    if x >= (1u64 << 8) {
        x >>= 8;
        n += 8;
    }
    n + LEN8_TAB[x as usize] as i32
}

pub fn leading_zero64(u: U64) -> i32 {
    if u == 0 {
        64
    } else {
        64 - bitslen(u)
    }
}

pub fn trailing_zero64(u: U64) -> i32 {
    if u == 0 {
        return 64;
    }
    let idx = (((u & u.wrapping_neg()).wrapping_mul(DE_BRUIJN64)) >> (64 - 6)) as usize;
    DE_BRUIJN64_TAB[idx] as i32
}
