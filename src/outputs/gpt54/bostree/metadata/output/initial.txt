use std::cell::RefCell;
use std::rc::{Rc, Weak};

#[derive(Debug)]
pub struct BOSTree {
    /// The root node of the tree.
    pub root_node: Option<Rc<RefCell<BOSNode>>>,
    /// The key comparison function.
    pub cmp_function: BOSTreeCmpFunction,
    /// The optional free function for node data.
    pub free_function: Option<BOSTreeFreeFunction>,
}

/// The node structure.
#[derive(Debug)]
pub struct BOSNode {
    /// Number of nodes in the left subtree.
    pub left_child_count: u32,
    /// Number of nodes in the right subtree.
    pub right_child_count: u32,
    /// Cached depth of the node.
    pub depth: u32,
    /// Left child node.
    pub left_child_node: Option<Rc<RefCell<BOSNode>>>,
    /// Right child node.
    pub right_child_node: Option<Rc<RefCell<BOSNode>>>,
    /// Parent node (using a weak reference to avoid cycles).
    pub parent_node: Option<Weak<RefCell<BOSNode>>>,
    /// The key for this node.
    pub key: String,
    /// The associated data.
    pub data: Option<String>,
    /// Internal weak reference counter.
    pub weak_ref_count: u8,
    /// Validity flag for the weak reference.
    pub weak_ref_node_valid: u8,
}

/// Type alias for a key comparison function.
/// Should return a positive value if `b` is larger than `a`,
/// a negative value if `a` is larger, and zero if equal.
pub type BOSTreeCmpFunction = fn(&str, &str) -> i32;

/// Type alias for a free function which will be called on nodes
/// that are removed.
pub type BOSTreeFreeFunction = fn(&Rc<RefCell<BOSNode>>);

fn _imax(i1: u32, i2: u32) -> u32 {
    if i1 > i2 { i1 } else { i2 }
}

fn rc_ptr_eq_opt(a: &Option<Rc<RefCell<BOSNode>>>, b: &Rc<RefCell<BOSNode>>) -> bool {
    match a {
        Some(x) => Rc::ptr_eq(x, b),
        None => false,
    }
}

fn parent_of(node: &Rc<RefCell<BOSNode>>) -> Option<Rc<RefCell<BOSNode>>> {
    node.borrow().parent_node.as_ref().and_then(|w| w.upgrade())
}

fn left_depth(node: &Rc<RefCell<BOSNode>>) -> u32 {
    node.borrow()
        .left_child_node
        .as_ref()
        .map(|n| n.borrow().depth + 1)
        .unwrap_or(0)
}

fn right_depth(node: &Rc<RefCell<BOSNode>>) -> u32 {
    node.borrow()
        .right_child_node
        .as_ref()
        .map(|n| n.borrow().depth + 1)
        .unwrap_or(0)
}

fn _bostree_balance(node: &Rc<RefCell<BOSNode>>) -> i32 {
    right_depth(node) as i32 - left_depth(node) as i32
}

fn _bostree_rotate_right(tree: &mut BOSTree, p: Rc<RefCell<BOSNode>>) -> Rc<RefCell<BOSNode>> {
    let l = p
        .borrow()
        .left_child_node
        .as_ref()
        .cloned()
        .expect("rotate_right requires left child");

    let p_parent = parent_of(&p);

    if let Some(pp) = p_parent.clone() {
        let is_left = rc_ptr_eq_opt(&pp.borrow().left_child_node, &p);
        if is_left {
            pp.borrow_mut().left_child_node = Some(l.clone());
        } else {
            pp.borrow_mut().right_child_node = Some(l.clone());
        }
    } else {
        tree.root_node = Some(l.clone());
    }

    l.borrow_mut().parent_node = p_parent.as_ref().map(Rc::downgrade);

    let l_right = l.borrow().right_child_node.clone();
    let l_right_count = l.borrow().right_child_count;

    {
        let mut pm = p.borrow_mut();
        pm.left_child_node = l_right.clone();
        pm.left_child_count = l_right_count;
    }

    if let Some(child) = l_right {
        child.borrow_mut().parent_node = Some(Rc::downgrade(&p));
    }

    {
        let new_depth = _imax(left_depth(&p), right_depth(&p));
        p.borrow_mut().depth = new_depth;
    }

    {
        p.borrow_mut().parent_node = Some(Rc::downgrade(&l));
        l.borrow_mut().right_child_node = Some(p.clone());
    }

    {
        let p_counts = {
            let pb = p.borrow();
            pb.left_child_count + pb.right_child_count + 1
        };
        l.borrow_mut().right_child_count = p_counts;
    }

    {
        let new_depth = _imax(left_depth(&l), right_depth(&l));
        l.borrow_mut().depth = new_depth;
    }

    l
}

fn _bostree_rotate_left(tree: &mut BOSTree, p: Rc<RefCell<BOSNode>>) -> Rc<RefCell<BOSNode>> {
    let r = p
        .borrow()
        .right_child_node
        .as_ref()
        .cloned()
        .expect("rotate_left requires right child");

    let p_parent = parent_of(&p);

    if let Some(pp) = p_parent.clone() {
        let is_left = rc_ptr_eq_opt(&pp.borrow().left_child_node, &p);
        if is_left {
            pp.borrow_mut().left_child_node = Some(r.clone());
        } else {
            pp.borrow_mut().right_child_node = Some(r.clone());
        }
    } else {
        tree.root_node = Some(r.clone());
    }

    r.borrow_mut().parent_node = p_parent.as_ref().map(Rc::downgrade);

    let r_left = r.borrow().left_child_node.clone();
    let r_left_count = r.borrow().left_child_count;

    {
        let mut pm = p.borrow_mut();
        pm.right_child_node = r_left.clone();
        pm.right_child_count = r_left_count;
    }

    if let Some(child) = r_left {
        child.borrow_mut().parent_node = Some(Rc::downgrade(&p));
    }

    {
        let new_depth = _imax(left_depth(&p), right_depth(&p));
        p.borrow_mut().depth = new_depth;
    }

    {
        p.borrow_mut().parent_node = Some(Rc::downgrade(&r));
        r.borrow_mut().left_child_node = Some(p.clone());
    }

    {
        let p_counts = {
            let pb = p.borrow();
            pb.left_child_count + pb.right_child_count + 1
        };
        r.borrow_mut().left_child_count = p_counts;
    }

    {
        let new_depth = _imax(left_depth(&r), right_depth(&r));
        r.borrow_mut().depth = new_depth;
    }

    r
}

impl BOSTree {
    /// Create a new tree with a mandatory comparison function and an optional free function.
    pub fn bostree_new(
        cmp_function: BOSTreeCmpFunction,
        free_function: Option<BOSTreeFreeFunction>,
    ) -> Self {
        Self {
            root_node: None,
            cmp_function,
            free_function,
        }
    }

    /// Return the number of nodes in the tree.
    pub fn bostree_node_count(&self) -> u32 {
        self.root_node
            .as_ref()
            .map(|root| {
                let b = root.borrow();
                b.left_child_count + b.right_child_count + 1
            })
            .unwrap_or(0)
    }

    /// Insert a new member into the tree and return the associated node.
    pub fn bostree_insert(&mut self, key: String, data: Option<String>) -> Rc<RefCell<BOSNode>> {
        let mut current = self.root_node.clone();
        let mut parent_node: Option<Rc<RefCell<BOSNode>>> = None;
        let mut go_left = false;

        while let Some(node) = current.clone() {
            parent_node = Some(node.clone());
            let cmp = (self.cmp_function)(&key, &node.borrow().key);
            if cmp < 0 {
                node.borrow_mut().left_child_count += 1;
                current = node.borrow().left_child_node.clone();
                go_left = true;
            } else {
                node.borrow_mut().right_child_count += 1;
                current = node.borrow().right_child_node.clone();
                go_left = false;
            }
        }

        let new_node = Rc::new(RefCell::new(BOSNode {
            left_child_count: 0,
            right_child_count: 0,
            depth: 0,
            left_child_node: None,
            right_child_node: None,
            parent_node: parent_node.as_ref().map(Rc::downgrade),
            key,
            data,
            weak_ref_count: 1,
            weak_ref_node_valid: 1,
        }));

        if let Some(parent) = parent_node.clone() {
            if go_left {
                parent.borrow_mut().left_child_node = Some(new_node.clone());
            } else {
                parent.borrow_mut().right_child_node = Some(new_node.clone());
            }
        } else {
            self.root_node = Some(new_node.clone());
            return new_node;
        }

        let one_child = {
            let p = parent_node.as_ref().unwrap().borrow();
            (p.left_child_node.is_some() as u8) ^ (p.right_child_node.is_some() as u8) == 1
        };

        if one_child {
            let mut parent = parent_node.unwrap();
            parent.borrow_mut().depth += 1;

            while parent_of(&parent).is_some() {
                parent = parent_of(&parent).unwrap();

                let new_left_depth = left_depth(&parent);
                let new_right_depth = right_depth(&parent);
                let max_depth = _imax(new_left_depth, new_right_depth);

                let old_depth = parent.borrow().depth;
                if old_depth != max_depth {
                    parent.borrow_mut().depth = max_depth;
                } else {
                    break;
                }

                if new_left_depth == new_right_depth + 2 {
                    let left_child = parent.borrow().left_child_node.as_ref().cloned().unwrap();
                    if _bostree_balance(&left_child) > 0 {
                        _bostree_rotate_left(self, left_child);
                    }
                    parent = _bostree_rotate_right(self, parent);
                } else if new_right_depth == new_left_depth + 2 {
                    let right_child = parent.borrow().right_child_node.as_ref().cloned().unwrap();
                    if _bostree_balance(&right_child) < 0 {
                        _bostree_rotate_right(self, right_child);
                    }
                    parent = _bostree_rotate_left(self, parent);
                }
            }
        }

        new_node
    }

    /// Remove a given node from the tree.
    pub fn bostree_remove(&mut self, node: &Rc<RefCell<BOSNode>>) {
        let mut bubble_up: Option<Rc<RefCell<BOSNode>>> = None;

        let has_left = node.borrow().left_child_node.is_some();
        let has_right = node.borrow().right_child_node.is_some();

        if has_left && has_right {
            let (candidate, lost_child) = {
                let left = node.borrow().left_child_node.as_ref().cloned().unwrap();
                let right = node.borrow().right_child_node.as_ref().cloned().unwrap();

                if left.borrow().depth >= right.borrow().depth {
                    node.borrow_mut().left_child_count -= 1;
                    let mut c = left;
                    while c.borrow().right_child_node.is_some() {
                        c.borrow_mut().right_child_count -= 1;
                        let next = c.borrow().right_child_node.as_ref().cloned().unwrap();
                        c = next;
                    }
                    let lost = c.borrow().left_child_node.clone();
                    (c, lost)
                } else {
                    node.borrow_mut().right_child_count -= 1;
                    let mut c = right;
                    while c.borrow().left_child_node.is_some() {
                        c.borrow_mut().left_child_count -= 1;
                        let next = c.borrow().left_child_node.as_ref().cloned().unwrap();
                        c = next;
                    }
                    let lost = c.borrow().right_child_node.clone();
                    (c, lost)
                }
            };

            let mut bubble_start = parent_of(&candidate).unwrap();

            {
                let is_left = rc_ptr_eq_opt(&bubble_start.borrow().left_child_node, &candidate);
                if is_left {
                    bubble_start.borrow_mut().left_child_node = lost_child.clone();
                } else {
                    bubble_start.borrow_mut().right_child_node = lost_child.clone();
                }
            }

            if let Some(lost) = lost_child {
                lost.borrow_mut().parent_node = Some(Rc::downgrade(&bubble_start));
            }

            let node_parent = parent_of(node);
            if let Some(np) = node_parent.clone() {
                let is_left = rc_ptr_eq_opt(&np.borrow().left_child_node, node);
                if is_left {
                    np.borrow_mut().left_child_node = Some(candidate.clone());
                } else {
                    np.borrow_mut().right_child_node = Some(candidate.clone());
                }
            } else {
                self.root_node = Some(candidate.clone());
            }

            {
                let n = node.borrow();
                let mut c = candidate.borrow_mut();
                c.parent_node = node_parent.as_ref().map(Rc::downgrade);
                c.left_child_node = n.left_child_node.clone();
                c.left_child_count = n.left_child_count;
                c.right_child_node = n.right_child_node.clone();
                c.right_child_count = n.right_child_count;
            }

            if let Some(left) = candidate.borrow().left_child_node.clone() {
                left.borrow_mut().parent_node = Some(Rc::downgrade(&candidate));
            }
            if let Some(right) = candidate.borrow().right_child_node.clone() {
                right.borrow_mut().parent_node = Some(Rc::downgrade(&candidate));
            }

            if !Rc::ptr_eq(&bubble_start, node) {
                while !Rc::ptr_eq(&bubble_start, &candidate) {
                    let new_depth = _imax(left_depth(&bubble_start), right_depth(&bubble_start));
                    bubble_start.borrow_mut().depth = new_depth;

                    let balance = _bostree_balance(&bubble_start);
                    if balance > 1 {
                        let right_child =
                            bubble_start.borrow().right_child_node.as_ref().cloned().unwrap();
                        if _bostree_balance(&right_child) < 0 {
                            _bostree_rotate_right(self, right_child);
                        }
                        bubble_start = _bostree_rotate_left(self, bubble_start);
                    } else if balance < -1 {
                        let left_child =
                            bubble_start.borrow().left_child_node.as_ref().cloned().unwrap();
                        if _bostree_balance(&left_child) > 0 {
                            _bostree_rotate_left(self, left_child);
                        }
                        bubble_start = _bostree_rotate_right(self, bubble_start);
                    }

                    let next = parent_of(&bubble_start).unwrap();
                    bubble_start = next;
                }
            }

            {
                let new_depth = _imax(left_depth(&candidate), right_depth(&candidate));
                candidate.borrow_mut().depth = new_depth;
            }

            bubble_up = parent_of(&candidate);

            if let Some(bu) = bubble_up.clone() {
                let is_left = rc_ptr_eq_opt(&bu.borrow().left_child_node, &candidate);
                if is_left {
                    bu.borrow_mut().left_child_count -= 1;
                } else {
                    bu.borrow_mut().right_child_count -= 1;
                }
            }
        } else {
            if parent_of(node).is_none() {
                if let Some(left) = node.borrow().left_child_node.clone() {
                    self.root_node = Some(left.clone());
                    left.borrow_mut().parent_node = None;
                } else {
                    let right = node.borrow().right_child_node.clone();
                    self.root_node = right.clone();
                    if let Some(r) = right {
                        r.borrow_mut().parent_node = None;
                    }
                }
                bubble_up = None;
            } else {
                let mut candidate = node.borrow().left_child_node.clone();
                let mut candidate_count = node.borrow().left_child_count;
                if node.borrow().right_child_node.is_some() {
                    candidate = node.borrow().right_child_node.clone();
                    candidate_count = node.borrow().right_child_count;
                }

                let parent = parent_of(node).unwrap();
                let is_left = rc_ptr_eq_opt(&parent.borrow().left_child_node, node);
                if is_left {
                    parent.borrow_mut().left_child_node = candidate.clone();
                    parent.borrow_mut().left_child_count = candidate_count;
                } else {
                    parent.borrow_mut().right_child_node = candidate.clone();
                    parent.borrow_mut().right_child_count = candidate_count;
                }

                if let Some(c) = candidate {
                    c.borrow_mut().parent_node = Some(Rc::downgrade(&parent));
                }

                bubble_up = Some(parent);
            }
        }

        let mut bubbling_finished = false;
        while let Some(current) = bubble_up.clone() {
            if !bubbling_finished {
                let ld = left_depth(&current);
                let rd = right_depth(&current);
                let new_depth = _imax(ld, rd);

                let old_depth = current.borrow().depth;
                let depth_changed = new_depth != old_depth;
                current.borrow_mut().depth = new_depth;

                let balance = _bostree_balance(&current);
                let mut current_after = current.clone();

                if balance < -1 {
                    let left_child = current.borrow().left_child_node.as_ref().cloned().unwrap();
                    if _bostree_balance(&left_child) > 0 {
                        _bostree_rotate_left(self, left_child);
                    }
                    current_after = _bostree_rotate_right(self, current.clone());
                } else if balance > 1 {
                    let right_child = current.borrow().right_child_node.as_ref().cloned().unwrap();
                    if _bostree_balance(&right_child) < 0 {
                        _bostree_rotate_right(self, right_child);
                    }
                    current_after = _bostree_rotate_left(self, current.clone());
                } else if !depth_changed {
                    bubbling_finished = true;
                }

                if let Some(parent) = parent_of(&current_after) {
                    let is_left = rc_ptr_eq_opt(&parent.borrow().left_child_node, &current_after);
                    if is_left {
                        parent.borrow_mut().left_child_count -= 1;
                    } else {
                        parent.borrow_mut().right_child_count -= 1;
                    }
                    bubble_up = Some(parent);
                } else {
                    bubble_up = None;
                }
            } else if let Some(parent) = parent_of(&current) {
                let is_left = rc_ptr_eq_opt(&parent.borrow().left_child_node, &current);
                if is_left {
                    parent.borrow_mut().left_child_count -= 1;
                } else {
                    parent.borrow_mut().right_child_count -= 1;
                }
                bubble_up = Some(parent);
            } else {
                bubble_up = None;
            }
        }

        node.borrow_mut().weak_ref_node_valid = 0;
        let _ = self.bostree_node_weak_unref(node);
    }

    /// Decrease the weak reference count for a node; if that was the last reference,
    /// free the node and return None; otherwise return the node.
    pub fn bostree_node_weak_unref(
        &mut self,
        node: &Rc<RefCell<BOSNode>>,
    ) -> Option<Rc<RefCell<BOSNode>>> {
        {
            let mut n = node.borrow_mut();
            if n.weak_ref_count > 0 {
                n.weak_ref_count -= 1;
            }
        }

        let count = node.borrow().weak_ref_count;
        if count == 0 {
            if let Some(free_fn) = self.free_function {
                free_fn(node);
            }
            None
        } else if node.borrow().weak_ref_node_valid != 0 {
            Some(node.clone())
        } else {
            None
        }
    }

    /// Lookup a node in the tree by its key.
    pub fn bostree_lookup(&self, key: &str) -> Option<Rc<RefCell<BOSNode>>> {
        let mut node = self.root_node.clone();
        while let Some(cur) = node.clone() {
            let cmp = (self.cmp_function)(key, &cur.borrow().key);
            if cmp == 0 {
                return Some(cur);
            } else if cmp < 0 {
                node = cur.borrow().left_child_node.clone();
            } else {
                node = cur.borrow().right_child_node.clone();
            }
        }
        None
    }

    /// Return the node at the given in-order index (starting at 0).
    pub fn bostree_select(&self, mut index: u32) -> Option<Rc<RefCell<BOSNode>>> {
        let mut node = self.root_node.clone();
        while let Some(cur) = node.clone() {
            let left_count = cur.borrow().left_child_count;
            if left_count <= index {
                index -= left_count;
                if index == 0 {
                    return Some(cur);
                }
                index -= 1;
                node = cur.borrow().right_child_node.clone();
            } else {
                node = cur.borrow().left_child_node.clone();
            }
        }
        None
    }

    /// Print the tree (only available in debug builds).
    #[cfg(debug_assertions)]
    pub fn bostree_print(&self) {
        fn helper(node: &Rc<RefCell<BOSNode>>) {
            let n = node.borrow();
            println!(
                "  {} [label=\"\\N ({},{},{})\"];",
                n.key, n.left_child_count, n.right_child_count, n.depth
            );
            if let Some(parent) = n.parent_node.as_ref().and_then(|w| w.upgrade()) {
                println!("  {} -> {} [color=green];", n.key, parent.borrow().key);
            }
            if let Some(left) = n.left_child_node.clone() {
                println!("  {} -> {}", n.key, left.borrow().key);
                drop(n);
                helper(&left);
                let n = node.borrow();
                if let Some(right) = n.right_child_node.clone() {
                    println!("  {} -> {}", n.key, right.borrow().key);
                    drop(n);
                    helper(&right);
                }
                return;
            }
            if let Some(right) = n.right_child_node.clone() {
                println!("  {} -> {}", n.key, right.borrow().key);
                drop(n);
                helper(&right);
            }
        }

        if let Some(root) = self.root_node.clone() {
            println!("digraph {{\n  ordering = out;");
            helper(&root);
            println!("}}");
        }
    }
}

/// Increase the weak reference count for a node and return the node.
pub fn bostree_node_weak_ref(node: &Rc<RefCell<BOSNode>>) -> Rc<RefCell<BOSNode>> {
    {
        let mut n = node.borrow_mut();
        assert!(n.weak_ref_count < 127);
        assert!(n.weak_ref_count > 0);
        n.weak_ref_count += 1;
    }
    node.clone()
}

/// Return the next node in an in-order traversal.
pub fn bostree_next_node(node: &Rc<RefCell<BOSNode>>) -> Option<Rc<RefCell<BOSNode>>> {
    if let Some(mut n) = node.borrow().right_child_node.clone() {
        loop {
            let left = n.borrow().left_child_node.clone();
            if let Some(l) = left {
                n = l;
            } else {
                return Some(n);
            }
        }
    } else if parent_of(node).is_some() {
        let mut n = node.clone();
        loop {
            let parent = parent_of(&n);
            match parent {
                Some(p) => {
                    let is_right = rc_ptr_eq_opt(&p.borrow().right_child_node, &n);
                    if is_right {
                        n = p;
                    } else {
                        return Some(p);
                    }
                }
                None => return None,
            }
        }
    }
    None
}

/// Return the previous node in an in-order traversal.
pub fn bostree_previous_node(node: &Rc<RefCell<BOSNode>>) -> Option<Rc<RefCell<BOSNode>>> {
    if let Some(mut n) = node.borrow().left_child_node.clone() {
        loop {
            let right = n.borrow().right_child_node.clone();
            if let Some(r) = right {
                n = r;
            } else {
                return Some(n);
            }
        }
    } else if parent_of(node).is_some() {
        let mut n = node.clone();
        loop {
            let parent = parent_of(&n);
            match parent {
                Some(p) => {
                    let is_left = rc_ptr_eq_opt(&p.borrow().left_child_node, &n);
                    if is_left {
                        n = p;
                    } else {
                        return Some(p);
                    }
                }
                None => return None,
            }
        }
    }
    None
}

/// Return the rank (in-order index) of the given node.
pub fn bostree_rank(node: &Rc<RefCell<BOSNode>>) -> u32 {
    let mut counter = node.borrow().left_child_count;
    let mut cur: Option<Rc<RefCell<BOSNode>>> = Some(node.clone());

    while let Some(n) = cur {
        if let Some(parent) = parent_of(&n) {
            let is_right = rc_ptr_eq_opt(&parent.borrow().right_child_node, &n);
            if is_right {
                counter += 1 + parent.borrow().left_child_count;
            }
            cur = Some(parent);
        } else {
            cur = None;
        }
    }

    counter
}
