pub fn viterbi_decode(out: &mut [u8], input: &[u16], len: u16) -> u32 {
    if len > 244 * 2 {
        eprintln!("Input size exceeds max history");
    }
    viterbi_reset();
    let mut pos = 0;
    let mut i = 0;
    while i < len as usize {
        let s0 = input[i];
        let s1 = input[i + 1];
