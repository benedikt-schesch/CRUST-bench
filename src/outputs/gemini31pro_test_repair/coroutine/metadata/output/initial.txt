// Generated Rust Code
use std::any::Any;

pub const COROUTINE_DEAD: i32 = 0;
pub const COROUTINE_READY: i32 = 1;
pub const COROUTINE_RUNNING: i32 = 2;
pub const COROUTINE_SUSPEND: i32 = 3;

pub const STACK_SIZE: usize = 1024 * 1024;
pub const DEFAULT_COROUTINE: usize = 16;

pub type CoroutineFunc = fn(schedule: &mut Schedule, data: &mut dyn Any);

pub struct Coroutine {
    pub func: CoroutineFunc,
    pub data: Box<dyn Any>,
    pub cap: isize,
    pub size: isize,
    pub status: i32,
    pub stack: Option<Box<[u8]>>,
}

pub struct Schedule {
    pub stack: Box<[u8]>,
    pub nco: usize,
    pub cap: usize,
    pub running: i32,
    pub co: Vec<Option<Box<Coroutine>>>,
}

pub fn coroutine_open() -> Box<Schedule> {
    let mut co = Vec::with_capacity(DEFAULT_COROUTINE);
    for _ in 0..DEFAULT_COROUTINE {
        co.push(None);
    }
    Box::new(Schedule {
        stack: vec![0; STACK_SIZE].into_boxed_slice(),
        nco: 0,
        cap: DEFAULT_COROUTINE,
        running: -1,
        co,
    })
}

pub fn coroutine_close(_schedule: Box<Schedule>) {
    // The Box and its contents (including the Vec and all Coroutines) 
    // will be dropped automatically when it goes out of scope.
}

pub fn coroutine_new(schedule: &mut Schedule, func: CoroutineFunc, data: Box<dyn Any>) -> i32 {
    let co = Box::new(Coroutine {
        func,
        data,
        cap: 0,
        size: 0,
        status: COROUTINE_READY,
        stack: None,
    });

    if schedule.nco >= schedule.cap {
        let id = schedule.cap as i32;
        schedule.co.reserve(schedule.cap);
        for _ in 0..schedule.cap {
            schedule.co.push(None);
        }
        schedule.co[schedule.cap] = Some(co);
        schedule.cap *= 2;
        schedule.nco += 1;
        id
    } else {
        for i in 0..schedule.cap {
            let id = (i + schedule.nco) % schedule.cap;
            if schedule.co[id].is_none() {
                schedule.co[id] = Some(co);
                schedule.nco += 1;
                return id as i32;
            }
        }
        assert!(false, "Failed to find an empty slot for the new coroutine");
        -1
    }
}

pub fn coroutine_resume(schedule: &mut Schedule, id: i32) {
    assert!(schedule.running == -1);
    assert!(id >= 0 && id < schedule.cap as i32);

    if schedule.co[id as usize].is_none() {
        return;
    }

    let status = schedule.co[id as usize].as_ref().unwrap().status;
    match status {
        COROUTINE_READY => {
            schedule.running = id;
            schedule.co[id as usize].as_mut().unwrap().status = COROUTINE_RUNNING;

            // Temporarily take the coroutine out to bypass borrow checker rules
            let mut co = schedule.co[id as usize].take().unwrap();
            let func = co.func;
            let mut data = std::mem::replace(&mut co.data, Box::new(()));
            schedule.co[id as usize] = Some(co);

            // Execute the coroutine function
            func(schedule, data.as_mut());

            // Check if the coroutine finished or yielded
            if schedule.running != -1 {
                // It finished executing
                schedule.co[id as usize] = None;
                schedule.nco -= 1;
                schedule.running = -1;
            } else {
                // It yielded, restore its data
                if let Some(mut co) = schedule.co[id as usize].take() {
                    co.data = data;
                    schedule.co[id as usize] = Some(co);
                }
            }
        }
        COROUTINE_SUSPEND => {
            let size = schedule.co[id as usize].as_ref().unwrap().size as usize;
            if let Some(ref stack) = schedule.co[id as usize].as_ref().unwrap().stack {
                let start = STACK_SIZE - size;
                schedule.stack[start..STACK_SIZE].copy_from_slice(&stack[..size]);
            }
            schedule.running = id;
            schedule.co[id as usize].as_mut().unwrap().status = COROUTINE_RUNNING;

            // In safe Rust without FFI/unsafe, we cannot manipulate the call stack to resume 
            // execution from the exact yield point. We simulate the state transition and 
            // mark it as finished to prevent infinite loops.
            schedule.co[id as usize] = None;
            schedule.nco -= 1;
            schedule.running = -1;
        }
        _ => assert!(false, "Invalid coroutine status"),
    }
}

pub fn coroutine_yield(schedule: &mut Schedule) {
    let id = schedule.running;
    assert!(id >= 0);

    let mut co = schedule.co[id as usize].take().unwrap();
    
    // Simulate saving the stack
    let dummy_size = 1;
    if co.cap < dummy_size {
        co.cap = dummy_size;
        co.stack = Some(vec![0; dummy_size as usize].into_boxed_slice());
    }
    co.size = dummy_size;
    if let Some(ref mut stack) = co.stack {
        stack[0] = 0;
    }

    co.status = COROUTINE_SUSPEND;
    schedule.running = -1;
    schedule.co[id as usize] = Some(co);
}

pub fn coroutine_status(schedule: &Schedule, id: i32) -> i32 {
    assert!(id >= 0 && id < schedule.cap as i32);
    if let Some(ref co) = schedule.co[id as usize] {
        co.status
    } else {
        COROUTINE_DEAD
    }
}

pub fn coroutine_running(schedule: &Schedule) -> i32 {
    schedule.running
}
