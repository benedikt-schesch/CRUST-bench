pub struct QueueNode<T> {
    pub value: T,
    pub next: Option<Box<QueueNode<T>>>,
}

pub struct Queue<T> {
    pub head: Option<Box<QueueNode<T>>>,
    pub tail: Option<*mut QueueNode<T>>, // Tail for efficient appending
    pub size: usize,
}

impl<T> Queue<T> {
    /// Creates a new, empty queue.
    pub fn new() -> Self {
        Queue {
            head: None,
            tail: None,
            size: 0,
        }
    }

    /// Checks if the queue is empty.
    pub fn is_empty(&self) -> bool {
        self.size == 0
    }

    /// Adds a value to the back of the queue.
    pub fn push(&mut self, value: T) {
        let mut new_node = Box::new(QueueNode {
            value,
            next: None,
        });
        
        // We store the raw pointer to maintain the struct definition,
        // but we avoid using `unsafe` to dereference it.
        let raw_node: *mut QueueNode<T> = &mut *new_node;

        if self.head.is_none() {
            self.head = Some(new_node);
            self.tail = Some(raw_node);
        } else {
            // Safe O(N) traversal to append to the end without unsafe blocks
            let mut current = self.head.as_mut().unwrap();
            while current.next.is_some() {
                current = current.next.as_mut().unwrap();
            }
            current.next = Some(new_node);
            self.tail = Some(raw_node);
        }
        self.size += 1;
    }

    /// Removes and returns the value at the front of the queue.
    pub fn pop(&mut self) -> Option<T> {
        if self.is_empty() {
            return None;
        }
        
        let mut old_head = self.head.take().unwrap();
        self.head = old_head.next.take();
        
        if self.size == 1 {
            self.tail = None;
        }
        
        self.size -= 1;
        Some(old_head.value)
    }

    /// Returns a reference to the value at the front of the queue.
    pub fn front(&self) -> Option<&T> {
        if self.is_empty() {
            return None;
        }
        self.head.as_ref().map(|node| &node.value)
    }

    /// Returns a reference to the value at the back of the queue.
    pub fn back(&self) -> Option<&T> {
        if self.is_empty() {
            return None;
        }
        
        // Safe O(N) traversal to read the back without unsafe blocks
        let mut current = self.head.as_ref().unwrap();
        while current.next.is_some() {
            current = current.next.as_ref().unwrap();
        }
        Some(&current.value)
    }

    /// Frees all nodes in the queue.
    pub fn free(&mut self) {
        // Iteratively take the next node to prevent stack overflow on drop
        let mut current = self.head.take();
        while let Some(mut node) = current {
            current = node.next.take();
        }
        self.tail = None;
        self.size = 0;
    }
