pub mod leftpad {
    pub fn leftpad(s: &str, len: usize, pad: char) -> String {
        if s.len() >= len {
            s.to_string()
        } else {
            let mut result = String::with_capacity(len);
            for _ in 0..(len - s.len()) {
                result.push(pad);
            }
            result.push_str(s);
            result
        }
    }
}
