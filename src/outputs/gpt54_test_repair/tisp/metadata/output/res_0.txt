    TspNone = 1 << 0,

    TspInt = 1 << 2,

fn pair_car(v: &Val) -> Val {

fn pair_cdr(v: &Val) -> Val {

pub fn rec_add(rec: &mut Rec, key: &str, val: Val) {
    if let Some(i) = rec.items.iter().position(|e| e.key == key) {
        rec.items[i].val = val;
        return;
    }
    rec.items.push(Entry {
        key: key.to_string(),
        val,
    });
    rec.size = rec.items.len() as i32;
}

pub fn mk_rat(num: i32, den: i32) -> Option<Val> {
    if den == 0 {
        return None;
    }
    let mut n = num;
    let mut d = den;
    frac_reduce(&mut n, &mut d);
    if d < 0 {
        d = -d;
        n = -n;
    }
    if d == 1 {
        return Some(mk_int(n));
    }
    Some(Val {
        t: TspType::TspRatio,
        v: ValUnion::N {
            num: n as f64,
            den: d as f64,
        },
    })
}

pub fn mk_val(t: TspType) -> Val {

pub fn tsp_lstlen(v: &Val) -> i32 {
    let mut len = 0;
    let mut cur = v.clone();
    while cur.t == TspType::TspPair {
        len += 1;
        cur = pair_cdr(&cur);
    }
    if is_nil(&cur) {
        len
    } else {
        -(len + 1)
    }
}

pub fn tisp_env_init(cap: usize) -> Tsp {
    let nil = mk_val(TspType::TspNil);
    let none = mk_val(TspType::TspNone);
    let t = Val {
        t: TspType::TspSym,
        v: ValUnion::S("True".to_string()),
    };
    let mut st = Tsp {
        file: String::new(),
        filec: 0,
        none: none.clone(),
        nil: nil.clone(),
        t: t.clone(),
        env: rec_new(cap, None),
        strs: rec_new(cap, None),
        syms: rec_new(cap, None),
        libh: Vec::new(),
        libhc: 0,
    };
    tisp_env_add(&mut st, "True", t);
    tisp_env_add(&mut st, "Nil", nil);
    tisp_env_add(&mut st, "Void", none);
    let bt_val = st.nil.clone();
    tisp_env_add(&mut st, "bt", bt_val);
    let version_val = mk_str(&mut st, "0.1").unwrap_or_else(none_val);
    tisp_env_add(&mut st, "version", version_val);
    st
}

pub fn frac_reduce(num: &mut i32, den: &mut i32) {
    let mut a = num.abs();
    let mut b = den.abs();
    if b == 0 {
        return;
    }
    let mut c = a % b;
    while c > 0 {
        a = b;
        b = c;
        c = a % b;
    }
    *num /= b;
    *den /= b;
}

pub fn isnum(s: &str) -> bool {
    let bytes = s.as_bytes();
    if bytes.is_empty() {
        return false;
    }
    let c0 = bytes[0] as char;
    c0.is_ascii_digit()
        || (c0 == '.'
            && bytes
                .get(1)
                .map(|c| (*c as char).is_ascii_digit())
                .unwrap_or(false))
        || ((c0 == '-' || c0 == '+')
            && bytes
                .get(1)
                .map(|c| {
                    let d = *c as char;
                    d.is_ascii_digit() || d == '.'
                })
                .unwrap_or(false))
}

pub fn esc_str(s: &str, len: i32, do_esc: i32) -> String {
    let mut out = String::new();
    let mut chars = s.chars().take(len as usize);
    while let Some(c) = chars.next() {
        if c == '\\' && do_esc != 0 {
            if let Some(n) = chars.next() {
                out.push(esc_char(n));
            }
        } else {
            out.push(c);
        }
    }
    out
}

pub fn skip_ws(st: &mut Tsp, skipnl: i32) {
    loop {
        if st.filec >= st.file.len() {
            break;
        }
        let rest = &st.file[st.filec..];
        let ws = if skipnl != 0 { " \t\n\r" } else { " \t" };
        let mut consumed = 0usize;
        for ch in rest.chars() {
            if ws.contains(ch) {
                consumed += ch.len_utf8();
            } else {
                break;
            }
        }
        st.filec += consumed;
        if st.filec >= st.file.len() {
            break;
        }
        let ch = st.file[st.filec..].chars().next().unwrap_or('\0');
        if ch == ';' {
            while st.filec < st.file.len() {
                let c = st.file[st.filec..].chars().next().unwrap_or('\0');
                st.filec += c.len_utf8();
                if c == '\n' {
                    if skipnl == 0 {
                        st.filec -= 1;
                    }
                    break;
                }
            }
            continue;
        }
        break;
    }
}

pub fn rec_extend(rec: &mut Rec, args: Val, vals: Val) -> Rec {
    let argnum = TSP_REC_FACTOR as i32 * tsp_lstlen(&args).abs().max(1);
    let mut ret = rec_new(argnum as usize, Some(Box::new(rec.clone())));
    let mut a = args;
    let mut v = vals;
    loop {
        if is_nil(&a) {
            break;
        }
        if a.t == TspType::TspPair {
            let arg = pair_car(&a);
            let val = if v.t == TspType::TspPair {
                pair_car(&v)
            } else {
                v.clone()
            };
            if arg.t == TspType::TspSym {
                rec_add(&mut ret, &get_s(&arg), val);
            }
            a = pair_cdr(&a);
            v = if v.t == TspType::TspPair {
                pair_cdr(&v)
            } else {
                nil_val()
            };
        } else {
            if a.t == TspType::TspSym {
                rec_add(&mut ret, &get_s(&a), v.clone());
            }
            break;
        }
    }
    ret
}

pub fn hash(key: &str) -> u32 {
    let mut h: u32 = 0;
    for b in key.bytes() {
        h = h.wrapping_mul(33).wrapping_add(b as u32);
    }
    h
}

fn peek_char(st: &Tsp) -> Option<char> {
    st.file[st.filec..].chars().next()
}

fn next_char(st: &mut Tsp) -> Option<char> {
    let ch = peek_char(st)?;
    st.filec += ch.len_utf8();
    Some(ch)
}

fn read_token(st: &mut Tsp) -> String {
    let mut out = String::new();
    while let Some(ch) = peek_char(st) {
        if ch.is_whitespace() || ch == '(' || ch == ')' {
            break;
        }
        out.push(ch);
        st.filec += ch.len_utf8();
    }
    out
}

fn parse_int_like(s: &str) -> Option<i32> {
    s.parse::<i32>().ok()
}

fn parse_number_token(tok: &str) -> Option<Val> {
    if tok.is_empty() {
        return None;
    }

    if tok.contains('/') {
        let mut parts = tok.splitn(2, '/');
        let left = parts.next()?;
        let right = parts.next()?;
        let n = if left.is_empty() || left == "+" || left == "-" {
            return None;
        } else {
            parse_int_like(left)?
        };
        let d = if right.is_empty() || right == "+" || right == "-" {
            return None;
        } else {
            parse_int_like(right)?
        };
        return mk_rat(n, d);
    }

    if tok.contains('.') || tok.contains('e') || tok.contains('E') {
        let v = tok.parse::<f64>().ok()?;
        return mk_dec(v);
    }

    parse_int_like(tok).map(mk_int)
}

pub fn tisp_read(st: &mut Tsp) -> Option<Val> {
    skip_ws(st, 1);
    let ch = peek_char(st)?;
    match ch {
        '(' => {
            let _ = next_char(st);
            skip_ws(st, 1);
            if peek_char(st) == Some(')') {
                let _ = next_char(st);
                Some(st.nil.clone())
            } else {
                None
            }
        }
        '"' => {
            let _ = next_char(st);
            let mut out = String::new();
            while let Some(c) = next_char(st) {
                if c == '"' {
                    return mk_str(st, &out);
                }
                if c == '\\' {
                    if let Some(n) = next_char(st) {
                        out.push(esc_char(n));
                    } else {
                        break;
                    }
                } else {
                    out.push(c);
                }
            }
            None
        }
        _ => {
            let tok = read_token(st);
            if tok.is_empty() {
                return None;
            }
            if let Some(v) = parse_number_token(&tok) {
                Some(v)
            } else {
                mk_sym(st, &tok)
            }
        }
    }
}

pub fn tisp_eval_body(st: &mut Tsp, env: &mut Rec, v: Val) -> Option<Val> {
    let mut cur = v;
    let mut ret = st.none.clone();
    while cur.t == TspType::TspPair {
        let expr = pair_car(&cur);
        if let Some(ev) = tisp_eval(st, expr) {
            ret = ev;
        } else {
            return None;
        }
        cur = pair_cdr(&cur);
    }
    let _ = env;
    Some(ret)
}

pub fn mk_list(st: &mut Tsp, n: i32, args: Vec<Val>) -> Option<Val> {
    if n <= 0 || args.is_empty() {
        return Some(st.nil.clone());
    }
    let mut out = st.nil.clone();
    for v in args.into_iter().take(n as usize).rev() {
        out = mk_pair(v, out)?;
    }
    Some(out)
}

pub fn vals_eq(a: &Val, b: &Val) -> bool {
    if ((a.t as u32) & TSP_NUM) != 0 && ((b.t as u32) & TSP_NUM) != 0 {
        let av = get_num(a) / get_den(a);
        let bv = get_num(b) / get_den(b);
        return av == bv;
    }
    if a.t != b.t {
        return false;
    }
    match (&a.v, &b.v) {
        (ValUnion::S(x), ValUnion::S(y)) => x == y,
        (
            ValUnion::N { num: xn, den: xd },
            ValUnion::N { num: yn, den: yd },
        ) => xn == yn && xd == yd,
        (
            ValUnion::P { car: ac, cdr: ad },
            ValUnion::P { car: bc, cdr: bd },
        ) => vals_eq(ac, bc) && vals_eq(ad, bd),
        (
            ValUnion::F { args: aa, body: ab, .. },
            ValUnion::F { args: ba, body: bb, .. },
        ) => vals_eq(aa, ba) && vals_eq(ab, bb),
        (ValUnion::R(_), ValUnion::R(_)) => false,
        _ => false,
    }
}

        '"' => '"',
        _ => c,
    }
}

pub fn tisp_print(f: &mut std::fs::File, v: &Val) {
    let out = match &v.v {
        ValUnion::S(s) => match v.t {
            TspType::TspStr | TspType::TspSym => s.clone(),
            TspType::TspNil => "Nil".to_string(),
            TspType::TspNone => "Void".to_string(),
            _ => s.clone(),
        },
        ValUnion::N { num, den } => match v.t {
            TspType::TspInt => (*num as i32).to_string(),
            TspType::TspRatio => {
                if *den == 1.0 {
                    (*num as i32).to_string()
                } else {
                    format!("{}/{}", *num as i32, *den as i32)
                }
            }
            TspType::TspDec => format_number_like_input(*num),
            _ => format_number_like_input(*num),
        },
        ValUnion::P { .. } => "Pair".to_string(),
        ValUnion::Pr { name, .. } => name.clone(),
        ValUnion::F { name, .. } => name.clone(),
        ValUnion::R(_) => "Rec".to_string(),
    };
    let _ = std::io::Write::write_all(f, out.as_bytes());
}
