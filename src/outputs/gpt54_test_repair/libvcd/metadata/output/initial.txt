pub fn isexpression(c: char) -> bool {
    match c {
        '-' | '0'..='9' | 'z' | 'Z' | 'x' | 'X' | 'b' | 'U' => true,
        _ => false,
    }
}

pub const VCD_SIGNAL_COUNT: usize = 32;
pub const VCD_VALUE_CHANGE_COUNT: usize = 4096;
pub const VCD_SIGNAL_SIZE: usize = 64;
pub const VCD_NAME_SIZE: usize = 32;
pub const VCD_TIME_UNIT_SIZE: usize = 8;
pub const VCD_VERSION_SIZE: usize = 64;
pub const VCD_DATE_SIZE: usize = 64;
pub type Timestamp = u32;

#[derive(Debug, Clone, Copy)]
pub struct ValueChange {
    pub timestamp: Timestamp,
    pub value: [u8; VCD_SIGNAL_SIZE],
}

#[derive(Debug)]
pub struct Signal {
    pub name: [u8; VCD_NAME_SIZE],
    pub size: usize,
    pub value_changes: Vec<ValueChange>,
}

#[derive(Debug)]
pub struct Timescale {
    pub unit: [u8; VCD_TIME_UNIT_SIZE],
    pub scale: usize,
}

#[derive(Debug)]
pub struct VCD {
    pub signals: Vec<Signal>,
    pub date: [u8; VCD_DATE_SIZE],
    pub version: [u8; VCD_VERSION_SIZE],
    pub timescale: Timescale,
}

fn zeroed_value_change() -> ValueChange {
    ValueChange {
        timestamp: 0,
        value: [0; VCD_SIGNAL_SIZE],
    }
}

fn write_c_string<const N: usize>(dst: &mut [u8; N], src: &str) {
    dst.fill(0);
    let bytes = src.as_bytes();
    let len = bytes.len().min(N.saturating_sub(1));
    dst[..len].copy_from_slice(&bytes[..len]);
}

fn read_c_string(bytes: &[u8]) -> String {
    let len = bytes.iter().position(|&b| b == 0).unwrap_or(bytes.len());
    String::from_utf8_lossy(&bytes[..len]).into_owned()
}

fn new_signal() -> Signal {
    Signal {
        name: [0; VCD_NAME_SIZE],
        size: 0,
        value_changes: Vec::new(),
    }
}

fn new_vcd() -> VCD {
    VCD {
        signals: Vec::new(),
        date: [0; VCD_DATE_SIZE],
        version: [0; VCD_VERSION_SIZE],
        timescale: Timescale {
            unit: [0; VCD_TIME_UNIT_SIZE],
            scale: 0,
        },
    }
}

struct Parser {
    chars: Vec<char>,
    pos: usize,
}

impl Parser {
    fn new(input: String) -> Self {
        Self {
            chars: input.chars().collect(),
            pos: 0,
        }
    }

    fn eof(&self) -> bool {
        self.pos >= self.chars.len()
    }

    fn peek(&self) -> Option<char> {
        self.chars.get(self.pos).copied()
    }

    fn next(&mut self) -> Option<char> {
        if self.eof() {
            None
        } else {
            let c = self.chars[self.pos];
            self.pos += 1;
            Some(c)
        }
    }

    fn skip_whitespace(&mut self) {
        while let Some(c) = self.peek() {
            if c.is_whitespace() {
                self.pos += 1;
            } else {
                break;
            }
        }
    }

    fn read_token(&mut self) -> Option<String> {
        self.skip_whitespace();
        if self.eof() {
            return None;
        }
        let start = self.pos;
        while let Some(c) = self.peek() {
            if c.is_whitespace() {
                break;
            }
            self.pos += 1;
        }
        Some(self.chars[start..self.pos].iter().collect())
    }

    fn read_until_newline(&mut self) -> String {
        let start = self.pos;
        while let Some(c) = self.peek() {
            if c == '\n' {
                break;
            }
            self.pos += 1;
        }
        self.chars[start..self.pos].iter().collect()
    }

    fn consume_newline(&mut self) {
        if self.peek() == Some('\n') {
            self.pos += 1;
        }
    }

    fn skip_until_dollar(&mut self) {
        while let Some(c) = self.peek() {
            if c == '$' {
                break;
            }
            self.pos += 1;
        }
    }
}

impl VCD {
    pub fn read_from_path(path: &str) -> Result<Self, std::io::Error> {
        let mut file = File::open(path)?;
        let mut content = String::new();
        file.read_to_string(&mut content)?;

        let mut parser = Parser::new(content);
        let mut vcd = new_vcd();
        let mut current_timestamp: Timestamp = 0;
        let mut state = State::BeforeModuleDefinitions;

        while let Some(character) = parser.next() {
            if character == '$' {
                if parse_instruction_impl(&mut parser, &mut vcd, &mut state).is_ok() {
                    continue;
                }
            } else if character == '#' {
                if let Ok(ts) = parse_timestamp_impl(&mut parser) {
                    current_timestamp = ts;
                    continue;
                }
            } else if isexpression(character) {
                parser.pos = parser.pos.saturating_sub(1);
                if parse_assignment_impl(&mut parser, &mut vcd, &current_timestamp).is_ok() {
                    continue;
                }
            } else if character.is_whitespace() {
                continue;
            }

            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                "failed to parse VCD input",
            ));
        }

        Ok(vcd)
    }

    pub fn get_signal_by_name(&self, signal_name: &str) -> Option<&Signal> {
        self.signals
            .iter()
            .find(|signal| read_c_string(&signal.name) == signal_name)
    }
}

impl Signal {
    pub fn get_value_at_timestamp(&self, timestamp: Timestamp) -> Option<&[u8; VCD_SIGNAL_SIZE]> {
        let mut previous_value: Option<&[u8; VCD_SIGNAL_SIZE]> = None;
        for value_change in &self.value_changes {
            if timestamp < value_change.timestamp {
                break;
            }
            previous_value = Some(&value_change.value);
        }
        previous_value
    }
}

pub const BUFFER_LENGTH: usize = 512;

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum State {
    BeforeModuleDefinitions,
    InsideTopModule,
    InsideInnerModules,
}

fn parse_instruction_impl(
    parser: &mut Parser,
    vcd: &mut VCD,
    state: &mut State,
) -> Result<(), std::io::Error> {
    let instruction = parser
        .read_token()
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing instruction"))?;

    if instruction == "end" || instruction == "dumpvars" || instruction == "dumpall" {
        return Ok(());
    }

    if instruction == "scope" {
        match *state {
            State::BeforeModuleDefinitions => *state = State::InsideTopModule,
            State::InsideTopModule => *state = State::InsideInnerModules,
            State::InsideInnerModules => {}
        }
        parser.consume_newline();
        parser.skip_until_dollar();
        return Ok(());
    }

    if instruction == "upscope"
        || instruction == "enddefinitions"
        || instruction == "comment"
    {
        parser.consume_newline();
        parser.skip_until_dollar();
        return Ok(());
    }

    if instruction == "var" {
        if *state == State::InsideInnerModules {
            parser.read_until_newline();
            parser.consume_newline();
            return Ok(());
        }

        let rest = parser.read_until_newline();
        parser.consume_newline();

        let trimmed = rest.trim();
        let mut parts = trimmed.split_whitespace();

        let _var_type = parts.next();
        let size_str = parts.next();
        let signal_id = parts.next();

        let mut name = String::new();
        if let Some(id) = signal_id {
            if let Some(idx) = trimmed.find(id) {
                let after_id = &trimmed[idx + id.len()..];
                let after_id = after_id.trim_start();
                if let Some(end_pos) = after_id.find('$') {
                    name = after_id[..end_pos].trim().to_string();
                } else {
                    name = after_id.trim().to_string();
                }
            }
        }

        let mut signal = new_signal();
        if let Some(s) = size_str {
            signal.size = s.parse::<usize>().unwrap_or(0);
        }
        write_c_string(&mut signal.name, &name);

        let index = signal_id.and_then(get_signal_index);
        let current_index = vcd.signals.len();
        if current_index < VCD_SIGNAL_COUNT {
            vcd.signals.push(signal);
        }

        if let Some(idx) = index {
            if idx < vcd.signals.len() && vcd.signals[idx].size != 0 {
                return Ok(());
            }
        }

        return Ok(());
    }

    if instruction == "date" {
        parser.consume_newline();
        let date = parser.read_until_newline();
        write_c_string(&mut vcd.date, date.trim_end());
        return Ok(());
    }

    if instruction == "version" {
        parser.consume_newline();
        let version = parser.read_until_newline();
        write_c_string(&mut vcd.version, version.trim_end());
        return Ok(());
    }

    if instruction == "timescale" {
        parser.consume_newline();
        let line = parser.read_until_newline();
        let trimmed = line.trim();
        let mut digits = String::new();
        let mut unit = String::new();
        for c in trimmed.chars() {
            if c.is_ascii_digit() && unit.is_empty() {
                digits.push(c);
            } else {
                unit.push(c);
            }
        }
        vcd.timescale.scale = digits.parse::<usize>().unwrap_or(0);
        write_c_string(&mut vcd.timescale.unit, unit.trim());
        return Ok(());
    }

    Err(io::Error::new(
        io::ErrorKind::InvalidData,
        "unknown instruction",
    ))
}

fn parse_timestamp_impl(parser: &mut Parser) -> Result<Timestamp, std::io::Error> {
    let token = parser
        .read_token()
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing timestamp"))?;
    token
        .parse::<u32>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "invalid timestamp"))
}

fn parse_assignment_impl(
    parser: &mut Parser,
    vcd: &mut VCD,
    timestamp: &Timestamp,
) -> Result<(), std::io::Error> {
    let mut buffer = parser.read_until_newline();
    if buffer.len() > BUFFER_LENGTH {
        buffer.truncate(BUFFER_LENGTH);
    }
    parser.consume_newline();

    if buffer.is_empty() {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "empty assignment",
        ));
    }

    let first = buffer.chars().next().unwrap_or('\0');
    let is_vector = !"01xXzZ".contains(first);

    let (value, signal_id) = if is_vector {
        let mut parts = buffer.splitn(2, ' ');
        let value = parts.next().unwrap_or("").trim().to_string();
        let signal_id = parts.next().unwrap_or("").trim().to_string();
        (value, signal_id)
    } else {
        let mut chars = buffer.chars();
        let value = chars.next().unwrap_or('\0').to_string();
        let signal_id: String = chars.collect::<String>().trim().to_string();
        (value, signal_id)
    };

    if value.is_empty() || signal_id.is_empty() {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "invalid assignment",
        ));
    }

    if signal_id.len() > 1 {
        return Ok(());
    }

    let index = match get_signal_index(&signal_id) {
        Some(i) => i,
        None => return Ok(()),
    };

    if index >= vcd.signals.len() {
        return Ok(());
    }

    if vcd.signals[index].value_changes.len() >= VCD_VALUE_CHANGE_COUNT {
        return Ok(());
    }

    let mut change = zeroed_value_change();
    change.timestamp = *timestamp;
    write_c_string(&mut change.value, &value);
    vcd.signals[index].value_changes.push(change);

    Ok(())
}

pub fn get_signal_index(s: &str) -> Option<usize> {
    let ch = s.chars().next()?;
    let id = (ch as i32) - ('!' as i32);
    if id < 0 || id as usize >= VCD_SIGNAL_COUNT {
        None
    } else {
        Some(id as usize)
    }
}
