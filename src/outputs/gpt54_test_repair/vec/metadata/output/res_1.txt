pub const VEC_VERSION: &str = "0.2.1";

pub fn vec_insert<T>(data: &mut Vec<T>, idx: usize, value: T) -> i32 {
    data.insert(idx, value);
    0
}

pub fn vec_splice<T>(data: &mut Vec<T>, start: usize, count: usize) {
    if count == 0 {
        return;
    }

    let end = start.saturating_add(count).min(data.len());
    if start >= end || start >= data.len() {
        return;
    }

    data.drain(start..end);
}

pub fn vec_swapsplice<T>(data: &mut Vec<T>, start: usize, count: usize) {
    if count == 0 || start >= data.len() {
        return;
    }

    let len = data.len();
    let remove_count = count.min(len - start);

    if remove_count == 0 {
        return;
    }

    let old_len = len;
    let new_len = old_len - remove_count;

    for i in 0..remove_count {
        let dst = start + i;
        let src = new_len + i;
        if dst >= new_len {
            break;
        }
        data.swap(dst, src);
    }

    data.truncate(new_len);
}

pub fn vec_reserve_po2<T>(data: &mut Vec<T>, n: usize) -> i32 {
    if n == 0 {
        return 0;
    }

    let mut n2 = 1usize;
    while n2 < n {
        n2 <<= 1;
    }

    vec_reserve(data, n2)
}

pub fn vec_expand<T>(data: &mut Vec<T>) -> i32 {
    if data.len() + 1 > data.capacity() {
        let new_capacity = if data.capacity() == 0 {
            1
        } else {
            data.capacity() << 1
        };
        vec_reserve(data, new_capacity);
    }
    0
}

pub fn vec_reserve<T>(data: &mut Vec<T>, n: usize) -> i32 {
    if n > data.capacity() {
        let additional = n.saturating_sub(data.len());
        data.reserve_exact(additional);
    }
    0
}

pub fn vec_compact<T>(data: &mut Vec<T>) -> i32 {
    if data.is_empty() {
        *data = Vec::new();
    } else {
        data.shrink_to_fit();
    }
    0
}

pub fn vec_swap<T>(data: &mut Vec<T>, idx1: usize, idx2: usize) {
    if idx1 == idx2 {
        return;
    }
    data.swap(idx1, idx2);
}
