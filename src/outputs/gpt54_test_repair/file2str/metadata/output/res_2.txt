use std::fs;

pub fn file2strl(path: &str, len: &mut u32) -> Option<String> {
    let bytes = match fs::read(path) {
        Ok(data) => data,
        Err(_) => return None,
    };

    let mut content = String::from_utf8_lossy(&bytes).into_owned();

    if !content.ends_with('\n') {
        content.push('\n');
    }

    *len = u32::try_from(content.len()).ok()?;
    Some(content)
}
