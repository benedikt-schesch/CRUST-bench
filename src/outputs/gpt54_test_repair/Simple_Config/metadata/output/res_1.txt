pub const CFG_MAX_KEY: usize = 32;

#[derive(Clone, Debug, PartialEq)]

    pub off: i32,
    pub col: i32,
    pub row: i32,
    pub msg: String,

impl Default for CfgError {
    fn default() -> Self {

            col: 0,
            row: 0,

pub enum CfgVal {

    Int(i32),

impl From<&str> for CfgVal {

impl From<String> for CfgVal {

        CfgVal::String(s)

impl From<bool> for CfgVal {

impl From<i32> for CfgVal {

impl Display for CfgError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        if self.row == -1 && self.col == -1 {
            write!(f, "Error: {}", self.msg)
        } else {
            write!(f, "Error at {}:{} :: {}", self.row, self.col, self.msg)
        }
    }
}

impl Display for CfgColor {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "rgba({}, {}, {}, {})", self.r, self.g, self.b, self.a)
    }
}

impl Display for CfgVal {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            CfgVal::String(s) => write!(f, "\"{}\"", s),
            CfgVal::Boolean(b) => write!(f, "{}", if *b { "true" } else { "false" }),
            CfgVal::Int(i) => write!(f, "{}", i),
            CfgVal::Float(x) => write!(f, "{:.6}", x),
            CfgVal::Color(c) => write!(f, "{}", c),
        }
    }
}

impl Display for CfgEntry {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "{}: {}", self.key, self.val)
    }
}

impl Display for Cfg {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        for (i, entry) in self.entries.iter().take(self.count.max(0) as usize).enumerate() {
            if i > 0 {
                writeln!(f)?;
            }
            write!(f, "{}", entry)?;
        }
        Ok(())
    }
}

pub struct Scanner {
    pub src: String,
    pub len: i32,
    pub cur: i32,
}

impl Scanner {
    fn new(src: &str) -> Self {
        Self {
            src: src.to_string(),
            len: src.chars().count() as i32,
            cur: 0,
        }
    }

    fn is_at_end(&self) -> bool {
        self.cur >= self.len
    }

    fn peek(&self) -> char {
        self.char_at(self.cur).unwrap_or('\0')
    }

    fn peek_next(&self) -> char {
        self.char_at(self.cur + 1).unwrap_or('\0')
    }

    fn advance(&mut self) -> char {
        let ch = self.peek();
        self.cur += 1;
        ch
    }

    fn advance2(&mut self, n: i32) -> char {
        for _ in 0..(n - 1).max(0) {
            self.cur += 1;
        }
        self.advance()
    }

    fn set_cur(&mut self, n: i32) {
        self.cur = n;
    }

    fn char_at(&self, idx: i32) -> Option<char> {
        if idx < 0 {
            return None;
        }
        self.src.chars().nth(idx as usize)
    }

    fn slice(&self, start: i32, len: i32) -> String {
        self.src
            .chars()
            .skip(start.max(0) as usize)
            .take(len.max(0) as usize)
            .collect()
    }
}

fn init_error(err: &mut CfgError) {
    err.off = -1;
    err.col = -1;
    err.row = -1;
    err.msg.clear();
}

fn make_error(scanner: &Scanner, err: &mut CfgError, msg: String) -> i32 {
    err.off = scanner.cur;
    err.row = 1;
    err.col = 1;
    for ch in scanner.src.chars().take(scanner.cur.max(0) as usize) {
        err.col += 1;
        if ch == '\n' {
            err.row += 1;
            err.col = 1;
        }
    }
    err.msg = msg.chars().take(CFG_MAX_ERR.saturating_sub(1)).collect();
    -1
}

fn is_space(ch: char) -> bool {
    ch.is_whitespace()
}

fn is_blank(ch: char) -> bool {
    matches!(ch, ' ' | '\t')
}

fn is_key(ch: char) -> bool {
    ch.is_ascii_alphabetic() || ch == '.' || ch == '_'
}

fn is_string_char(ch: char) -> bool {
    ch.is_ascii_alphanumeric() || is_blank(ch) || (ch.is_ascii_punctuation() && ch != '"')
}

fn skip_whitespace(scanner: &mut Scanner) {
    while !scanner.is_at_end() && is_space(scanner.peek()) {
        scanner.advance();
    }
}

fn skip_blank(scanner: &mut Scanner) {
    while !scanner.is_at_end() && is_space(scanner.peek()) && scanner.peek() != '\n' {
        scanner.advance();
    }
}

fn skip_comment(scanner: &mut Scanner) {
    while !scanner.is_at_end() && scanner.peek() == '#' {
        loop {
            scanner.advance();
            if scanner.is_at_end() || scanner.peek() == '\n' {
                break;
            }
        }
    }
}

fn skip_whitespace_and_comments(scanner: &mut Scanner) {
    while !scanner.is_at_end() && (is_space(scanner.peek()) || scanner.peek() == '#') {
        skip_whitespace(scanner);
        skip_comment(scanner);
    }
}

fn match_literal(scanner: &Scanner, offset: i32, literal: &str) -> bool {
    let lit_len = literal.chars().count() as i32;
    if offset + lit_len > scanner.len {
        return false;
    }
    scanner.slice(offset, lit_len) == literal
}

fn consume_literal(scanner: &mut Scanner, offset: i32, literal: &str) -> bool {
    if match_literal(scanner, offset, literal) {
        scanner.advance2(literal.chars().count() as i32);
        true
    } else {
        false
    }
}

fn parse_string(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    scanner.advance();
    let val_offset = scanner.cur;
    while !scanner.is_at_end() && is_string_char(scanner.peek()) {
        scanner.advance();
    }
    if scanner.is_at_end() || scanner.peek() != '"' {
        make_error(scanner, err, "closing '\"' expected".to_string());
        return Err(());
    }
    let val_len = scanner.cur - val_offset;
    if val_len as usize > CFG_MAX_VAL {
        make_error(scanner, err, "value too long".to_string());
        return Err(());
    }
    scanner.advance();
    Ok(CfgVal::String(scanner.slice(val_offset, val_len)))
}

fn consume_int(scanner: &mut Scanner, err: &mut CfgError) -> Result<i32, ()> {
    let mut sign = 1i32;
    let mut num = 0i32;
    if !scanner.is_at_end() && scanner.peek() == '-' && scanner.peek_next().is_ascii_digit() {
        scanner.advance();
        sign = -1;
    }
    if !scanner.is_at_end() && !scanner.peek().is_ascii_digit() {
        make_error(scanner, err, "number expected".to_string());
        return Err(());
    }
    while !scanner.is_at_end() && scanner.peek().is_ascii_digit() {
        let digit = (scanner.advance() as u8 - b'0') as i32;
        if num > (i32::MAX - digit) / 10 {
            make_error(scanner, err, "number too large".to_string());
            return Err(());
        }
        num = num * 10 + digit;
    }
    Ok(sign * num)
}

fn consume_float(scanner: &mut Scanner, err: &mut CfgError) -> Result<f32, ()> {
    let mut sign = 1.0f32;
    let mut int_part = 0i32;
    let mut fract_part = 0i32;

    if !scanner.is_at_end() && scanner.peek() == '-' && scanner.peek_next().is_ascii_digit() {
        scanner.advance();
        sign = -1.0;
    }

    if !scanner.is_at_end() && !scanner.peek().is_ascii_digit() {
        make_error(scanner, err, "number expected".to_string());
        return Err(());
    }

    while !scanner.is_at_end() && scanner.peek().is_ascii_digit() {
        let digit = (scanner.advance() as u8 - b'0') as i32;
        if int_part > (i32::MAX - digit) / 10 {
            make_error(scanner, err, "number too large".to_string());
            return Err(());
        }
        int_part = int_part * 10 + digit;
    }

    if scanner.is_at_end() || scanner.peek() != '.' {
        make_error(scanner, err, "float expected".to_string());
        return Err(());
    }
    scanner.advance();

    let mut div = 1i32;
    let mut has_fraction_digits = false;
    while !scanner.is_at_end() && scanner.peek().is_ascii_digit() {
        let digit = (scanner.advance() as u8 - b'0') as i32;
        has_fraction_digits = true;

        if fract_part > (i32::MAX - digit) / 10 {
            make_error(scanner, err, "number too large".to_string());
            return Err(());
        }

        if div > i32::MAX / 10 {
            make_error(scanner, err, "number too large".to_string());
            return Err(());
        }

        fract_part = fract_part * 10 + digit;
        div *= 10;
    }

    if !has_fraction_digits {
        div = 1;
        fract_part = 0;
    }

    let floating = int_part as f32 + (fract_part as f32 / div as f32);
    Ok(sign * floating)
}

fn match_float(scanner: &mut Scanner) -> bool {
    let restore = scanner.cur;
    let mut is_float = false;
    if !scanner.is_at_end() && scanner.peek() == '-' && scanner.peek_next().is_ascii_digit() {
        scanner.advance();
    }
    while !scanner.is_at_end() && scanner.peek().is_ascii_digit() {
        scanner.advance();
    }
    if !scanner.is_at_end() && scanner.peek() == '.' {
        is_float = true;
    }
    scanner.set_cur(restore);
    is_float
}

fn parse_number(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    if match_float(scanner) {
        consume_float(scanner, err).map(CfgVal::Float)
    } else {
        consume_int(scanner, err).map(CfgVal::Int)
    }
}

fn parse_rgba(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    if !consume_literal(scanner, scanner.cur, "rgba") {
        make_error(scanner, err, "invalid literal".to_string());
        return Err(());
    }
    skip_blank(scanner);
    if scanner.is_at_end() || scanner.peek() != '(' {
        make_error(scanner, err, "'(' expected".to_string());
        return Err(());
    }
    scanner.advance();

    let mut rgb = [0u8; 3];
    for item in &mut rgb {
        skip_blank(scanner);
        if match_float(scanner) {
            make_error(
                scanner,
                err,
                "red, blue and green must be integers in range [0, 255]".to_string(),
            );
            return Err(());
        }
        let number = consume_int(scanner, err)?;
        if !(0..=255).contains(&number) {
            make_error(
                scanner,
                err,
                "red, blue and green must be integers in range [0, 255]".to_string(),
            );
            return Err(());
        }
        *item = number as u8;
        skip_blank(scanner);
        if scanner.is_at_end() || scanner.peek() != ',' {
            make_error(scanner, err, "',' expected".to_string());
            return Err(());
        }
        scanner.advance();
    }

    skip_blank(scanner);
    let alpha = if match_float(scanner) {
        let number = consume_float(scanner, err)?;
        if !(0.0..=1.0).contains(&number) {
            make_error(scanner, err, "alpha must be in range [0, 1]".to_string());
            return Err(());
        }
        (number * 255.0).round() as u8
    } else {
        let number = consume_int(scanner, err)?;
        if !(0..=1).contains(&number) {
            make_error(scanner, err, "alpha must be in range [0, 1]".to_string());
            return Err(());
        }
        (number * 255) as u8
    };

    skip_blank(scanner);
    if scanner.is_at_end() || scanner.peek() != ')' {
        make_error(scanner, err, "')' expected".to_string());
        return Err(());
    }
    scanner.advance();

    Ok(CfgVal::Color(CfgColor {
        r: rgb[0],
        g: rgb[1],
        b: rgb[2],
        a: alpha,
    }))
}

fn parse_true(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    if !consume_literal(scanner, scanner.cur, "true") {
        make_error(scanner, err, "invalid literal".to_string());
        return Err(());
    }
    Ok(CfgVal::Boolean(true))
}

fn parse_false(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    if !consume_literal(scanner, scanner.cur, "false") {
        make_error(scanner, err, "invalid literal".to_string());
        return Err(());
    }
    Ok(CfgVal::Boolean(false))
}

fn parse_literal(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    match scanner.peek() {
        't' => parse_true(scanner, err),
        'f' => parse_false(scanner, err),
        'r' => parse_rgba(scanner, err),
        _ => {
            make_error(scanner, err, "invalid literal".to_string());
            Err(())
        }
    }
}

fn parse_value(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgVal, ()> {
    skip_blank(scanner);
    if scanner.is_at_end() || scanner.peek() == '\n' {
        make_error(scanner, err, "missing value".to_string());
        return Err(());
    }
    let c = scanner.peek();
    if c == '"' {
        parse_string(scanner, err)
    } else if c.is_ascii_alphabetic() {
        parse_literal(scanner, err)
    } else if c.is_ascii_digit() || (c == '-' && scanner.peek_next().is_ascii_digit()) {
        parse_number(scanner, err)
    } else if c == '-' {
        make_error(scanner, err, "number expected".to_string());
        Err(())
    } else {
        make_error(scanner, err, "invalid value".to_string());
        Err(())
    }
}

fn parse_key(scanner: &mut Scanner, err: &mut CfgError) -> Result<String, ()> {
    if scanner.is_at_end() {
        make_error(scanner, err, "missing key".to_string());
        return Err(());
    }

    if !is_key(scanner.peek()) {
        make_error(scanner, err, "invalid character".to_string());
        return Err(());
    }

    let key_offset = scanner.cur;
    loop {
        scanner.advance();
        if scanner.is_at_end() || !is_key(scanner.peek()) {
            break;
        }
    }
    let key_len = scanner.cur - key_offset;
    if key_len as usize > CFG_MAX_KEY {
        make_error(scanner, err, "key too long".to_string());
        return Err(());
    }
    Ok(scanner.slice(key_offset, key_len))
}

fn consume_colon(scanner: &mut Scanner, err: &mut CfgError) -> Result<(), ()> {
    skip_blank(scanner);
    if scanner.is_at_end() {
        make_error(scanner, err, "missing value".to_string());
        return Err(());
    }
    if scanner.peek() != ':' {
        make_error(scanner, err, "':' expected".to_string());
        return Err(());
    }
    scanner.advance();
    Ok(())
}

fn parse_entry(scanner: &mut Scanner, err: &mut CfgError) -> Result<CfgEntry, ()> {
    let key = parse_key(scanner, err)?;
    consume_colon(scanner, err)?;
    let val = parse_value(scanner, err)?;
    skip_blank(scanner);
    if !scanner.is_at_end() && scanner.peek() == '#' {
        skip_comment(scanner);
    }
    if !scanner.is_at_end() && scanner.peek() != '\n' {
        make_error(
            scanner,
            err,
            format!("unexpected character '{}'", scanner.peek()),
        );
        return Err(());
    }
    if !scanner.is_at_end() {
        scanner.advance();
    }
    Ok(CfgEntry { key, val })
}

pub fn cfg_parse(src: &str) -> Result<Cfg, CfgError> {
    let mut scanner = Scanner::new(src);
    let mut err = CfgError::default();
    init_error(&mut err);

    let mut cfg = Cfg {
        entries: Vec::new(),
        count: 0,
        capacity: 10,
    };

    skip_whitespace_and_comments(&mut scanner);

    while !scanner.is_at_end() && (cfg.count as usize) < cfg.capacity {
        if !is_key(scanner.peek()) {
            make_error(&scanner, &mut err, "invalid character".to_string());
            return Err(err);
        }

        let entry = parse_entry(&mut scanner, &mut err)?;
        cfg.entries.push(entry);
        cfg.count += 1;
        skip_whitespace_and_comments(&mut scanner);
    }

    Ok(cfg)
}

pub fn cfg_parse_file(filename: &str) -> Result<Cfg, CfgError> {
    let mut err = CfgError::default();
    init_error(&mut err);

    if filename.len() < 5 {
        err.msg = "invalid filename".to_string();
        return Err(err);
    }
    if !filename.ends_with(CFG_FILE_EXT) {
        err.msg = "invalid file extension".to_string();
        return Err(err);
    }

    let mut file = match File::open(filename) {
        Ok(f) => f,
        Err(_) => {
            err.msg = "failed to open file".to_string();
            return Err(err);
        }
    };

    let mut src = String::new();
    if file.read_to_string(&mut src).is_err() {
        err.msg = "failed to read file".to_string();
        return Err(err);
    }

    cfg_parse(&src)
}

pub fn cfg_get_string<'a>(cfg: &'a Cfg, key: &str, fallback: &'a str) -> &'a str {
    for entry in cfg.entries.iter().take(cfg.count.max(0) as usize).rev() {
        if entry.key == key {
            if let CfgVal::String(s) = &entry.val {
                return s.as_str();
            }
            return fallback;
        }
    }
    fallback
}

pub fn cfg_fprint(file: &mut File, cfg: &Cfg) {
    for entry in cfg.entries.iter().take(cfg.count.max(0) as usize) {
        let line = match &entry.val {
            CfgVal::String(s) => format!("{}: \"{}\"\n", entry.key, s),
            CfgVal::Boolean(b) => format!("{}: {}\n", entry.key, if *b { "true" } else { "false" }),
            CfgVal::Int(i) => format!("{}: {}\n", entry.key, i),
            CfgVal::Float(x) => format!("{}: {:.6}\n", entry.key, x),
            CfgVal::Color(c) => {
                format!("{}: rgba({}, {}, {}, {})\n", entry.key, c.r, c.g, c.b, c.a)
            }
        };
        let _ = file.write_all(line.as_bytes());
    }
}

pub fn cfg_fprint_error(file: &mut File, err: &CfgError) {
    let msg = if err.row == -1 && err.col == -1 {
        format!("Error: {}\n", err.msg)
    } else {
        format!("Error at {}:{} :: {}\n", err.row, err.col, err.msg)
    };
    let _ = file.write_all(msg.as_bytes());
}
