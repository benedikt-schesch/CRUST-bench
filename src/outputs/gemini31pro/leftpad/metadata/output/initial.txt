pub fn leftpad(str: &str, padding: &str, min_len: usize, dest: &mut [u8]) -> usize {
    let str_bytes = str.as_bytes();
    let mut str_len = 0;
    while str_len < str_bytes.len() && str_bytes[str_len] != 0 {
        str_len += 1;
    }

    let pad_bytes = padding.as_bytes();
    let mut pad_len = 0;
    while pad_len < pad_bytes.len() && pad_bytes[pad_len] != 0 {
        pad_len += 1;
    }

    let actual_padding = if pad_len == 0 {
        b" "
    } else {
        &pad_bytes[..pad_len]
    };

    let mut npad = 0;
    if str_len < min_len {
        npad = min_len - str_len;
    }

    let dest_sz = dest.len();
    if dest_sz == 0 {
        return str_len + npad;
    }

    let mut dest_len = 0;
    let mut i = 0;
    while dest_len < npad && dest_len < dest_sz - 1 {
        dest[dest_len] = actual_padding[i];
        dest_len += 1;
        i += 1;
        if i == actual_padding.len() {
            i = 0;
        }
    }

    i = 0;
    while i < str_len && dest_len < dest_sz - 1 {
        dest[dest_len] = str_bytes[i];
        dest_len += 1;
        i += 1;
    }

    dest[dest_len] = b'\0';
    dest_len
}
