use std::hash::{Hash, Hasher};
use std::collections::hash_map::DefaultHasher;
use std::time::{SystemTime, UNIX_EPOCH};

pub const RANDOM_SIZE: usize = 16;

pub fn generate_random_data(buffer: &mut [u8]) {
    let charset = b"0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
    let mut seed = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_nanos() as u64;
    for i in 0..buffer.len() {
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
        buffer[i] = charset[(seed as usize) % charset.len()];
    }
}

pub fn generate_hash(data: &[u8], hash: &mut [u8]) {
    let mut hasher = DefaultHasher::new();
    data.hash(&mut hasher);
    let h1 = hasher.finish();
    hasher.write_u64(h1);
    let h2 = hasher.finish();
    hasher.write_u64(h2);
    let h3 = hasher.finish();
    hasher.write_u64(h3);
    let h4 = hasher.finish();
    
    let bytes = [
        h1.to_le_bytes(),
        h2.to_le_bytes(),
        h3.to_le_bytes(),
        h4.to_le_bytes(),
    ];
    
    for i in 0..32 {
        if i < hash.len() {
            hash[i] = bytes[i / 8][i % 8];
        }
    }
}
