file: &File,
    vcd: &mut VCD,
    timestamp: &Timestamp,
) -> Result<(), std::io::Error> {
    let mut buffer = String::new();
    let mut buf = [0u8; 1];
    let mut mut_file = file;
    while mut_file.read(&mut buf)? == 1 {
        let c = buf[0] as char;
        if c == '\n' {
            mut_file.seek(SeekFrom::Current(-1))?;
            break;
        }
        buffer.push(c);
    }
    
    if buffer.is_empty() {
        return Err(std::io::Error::new(std::io::ErrorKind::InvalidData, "Empty assignment"));
    }
    
    let first_char = buffer.chars().next().unwrap();
    let is_vector = !matches!(first_char, '0' | '1' | 'x' | 'X' | 'z' | 'Z');
    
    let (value_str, signal_id_str) = if is_vector {
        let parts: Vec<&str> = buffer.splitn(2, ' ').collect();
        if parts.len() != 2 {
            return Err(std::io::Error::new(std::io::ErrorKind::InvalidData, "Invalid vector assignment"));
        }
        (parts[0].to_string(), parts[1].trim_start().to_string())
    } else {
        let (v, s) = buffer.split_at(1);
        (v.to_string(), s.to_string())
    };
    
    if signal_id_str.is_empty() {
        return Err(std::io::Error::new(std::io::ErrorKind::InvalidData, "Missing signal id"));
    }
    
    if signal_id_str.len() > 1 {
        return Ok(());
    }
    
    let index = match get_signal_index(&signal_id_str) {
        Some(idx) => idx,
        None => return Ok(()),
    };
    
    if index >= vcd.signals.len() {
        return Ok(());
    }
    
    let mut value_bytes = [0u8; VCD_SIGNAL_SIZE];
    let bytes = value_str.as_bytes();
    let len = bytes.len().min(VCD_SIGNAL_SIZE - 1);
    value_bytes[..len].copy_from_slice(&bytes[..len]);
    
    vcd.signals[index].value_changes.push(ValueChange {
        timestamp: *timestamp,
        value: value_bytes,
    });
    
    Ok(())
}
