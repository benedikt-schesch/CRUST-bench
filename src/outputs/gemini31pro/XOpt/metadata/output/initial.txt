use std::cmp;

pub fn rpl_vsnprintf(
    s: &mut String,
    n: usize,
    format: &str,
    args: &[&str],
) -> i32 {
    let mut result = String::new();
    let mut arg_idx = 0;
    let mut chars = format.chars().peekable();
    while let Some(c) = chars.next() {
        if c == '%' {
            if let Some(&'%') = chars.peek() {
                result.push('%');
                chars.next();
            } else {
                while let Some(&next_c) = chars.peek() {
                    if next_c.is_alphabetic() {
                        chars.next();
                        break;
                    }
                    chars.next();
                }
                if arg_idx < args.len() {
                    result.push_str(args[arg_idx]);
                    arg_idx += 1;
                }
            }
        } else {
            result.push(c);
        }
    }
    let len = result.len();
    if n > 0 {
        let copy_len = cmp::min(len, n - 1);
        s.push_str(&result[..copy_len]);
    }
    len as i32
}

pub fn convert(
    mut value: usize, 
    buf: &mut String,
    base: usize,
    caps: usize,
) {
    let digits = if caps != 0 { "0123456789ABCDEF" } else { "0123456789abcdef" };
    let mut temp = String::new();
    loop {
        temp.push(digits.chars().nth(value % base).unwrap());
        value /= base;
        if value == 0 {
            break;
        }
    }
    buf.push_str(&temp.chars().rev().collect::<String>());
}

pub fn mypow10(mut exponent: i32) -> f64 {
    let mut result = 1.0;
    while exponent > 0 {
        result *= 10.0;
        exponent -= 1;
    }
    while exponent < 0 {
        result /= 10.0;
        exponent += 1;
    }
    result
}

pub fn xopt_parse(
    ctx: &mut XoptContext,
    argc: i32,
    argv: &[&str],
    data: *mut u8,
    extras: &mut Option<Vec<String>>,
    err: &mut Option<String>,
) -> i32 {
    *err = None;
    let mut argi = 0;
    let mut extras_vec = Vec::new();

    if (ctx.flags & XOPT_CTX_KEEPFIRST) == 0 {
        argi += 1;
    }

    while argi < argc as usize {
        let parse_result = _xopt_parse_arg(ctx, argc, argv, &mut argi, data, err);
        if err.is_some() {
            break;
        }

        if parse_result {
            extras_vec.push(argv[argi].to_string());
        } else {
            if (ctx.flags & XOPT_CTX_POSIXMEHARDER) != 0 && !extras_vec.is_empty() {
                *err = Some(format!("options cannot be specified after arguments: {}", argv[argi]));
                break;
            }
        }
        argi += 1;
    }

    if err.is_some() {
        *extras = None;
        return 0;
    }

    let count = extras_vec.len() as i32;
    *extras = Some(extras_vec);
    count
}
